035/**
036 * <p>
037 * This abstract class defines the core features for the execution of Service
038 * inside a BusinessWorks engine.
039 * </p>
040 * 
041 * @author Radwane Mabchour
042 *
043 */
044public abstract class AbstractServiceEngineMojo extends AbstractBWMojo {
045
046        protected final static String BWENGINE_MISSING = "The BusinessWorks engine can't be found.";
047        protected final static String BWENGINE_STARTING = "Starting BusinessWorks Engine";
048        protected final static String BWENGINE_STOPPING = "Stopping BusinessWorks Engine";
049        protected final static String BWENGINE_ERROR_STARTING = "Error while starting the BusinessWorks Engine";
050        protected final static String BWENGINE_TIMEOUT_STARTING = "Timeout while starting the BusinessWorks Engine";
051        
052        /**
053         * Path to the BusinessWorks Engine binary.
054         */
055        @Parameter (property = "bwengine.path", required = true)
056        protected File tibcoBWEnginePath;
057
058        /**
059         * <p>
060         * The minimum value for the randomly chosen port.
061         * </p>
062         */
063        @Parameter (property = "bw.service.minPort", defaultValue="49152")
064        protected int minPort;
065
066        /**
067         * <p>
068         * The maximum value for the randomly chosen port.
069         * </p>
070         */
071        @Parameter (property = "bw.service.maxPort", defaultValue="65535")
072        protected int maxPort;
073
074        /**
075         * A Service Agent is a Web Service representation in TIBCO BusinessWorks.
076         * This Service Agent will be started when the BusinessWorks engine starts.
077         */
078        protected ServiceAgentInEngine<?> serviceAgent;
079
080        private void bwEngineNotFound() throws MojoExecutionException {
081                throw new BinaryMissingException(BWENGINE_MISSING);
082        }
083
084        private void checkBWEngine() throws MojoExecutionException {
085                if (tibcoBWEnginePath == null ||
086                   !tibcoBWEnginePath.exists() ||
087                   !tibcoBWEnginePath.isFile()) {
088                        bwEngineNotFound();
089                }
090        }
091
092        /**
093         * 
094         * @return true si le Service Agent répond à l'invocation de la méthode
095         * isStarted qui agit comme une méthode de type "ping" pour savoir si le
096         * service est bien lancé, false sinon.
097         */
098        public boolean isStarted() {
099                return (serviceAgent == null) ? false : serviceAgent.isStarted();
100        }
101
102        /**
103         * This will call the stopEngine() method of the Service Agent. The Service
104         * Agent will then stop the BusinessWorks engine.
105         */
106        public void stopEngine() {
107                if (serviceAgent != null) {
108                        serviceAgent.stopEngine();
109                }
110        }
111
112        public abstract void initServiceAgent() throws MojoExecutionException;
113        
114        public abstract void executeServiceMethods() throws MojoExecutionException;
115
116        public abstract String getBWEnginePort();
117        public abstract int getRetryInterval();
118        public abstract String getServiceName();
119        public abstract String getServiceFailureMessage();
120        public abstract String getServiceTimeoutMessage();
121        
122        public int getTimeOut() {
123                return timeOut; // default TimeOut, concrete children will override
124        }
125        
126        private int getMaxRetry() {
127                return getTimeOut() / getRetryInterval();
128        }
129
130        public static boolean available(int port, int minPort, int maxPort) {
131            if (port < minPort || port > maxPort || port > 65535) {
132                throw new IllegalArgumentException("Invalid start port: " + port);
133            }
134
135            ServerSocket ss = null;
136            DatagramSocket ds = null;
137            try {
138                ss = new ServerSocket(port);
139                ss.setReuseAddress(true);
140                ds = new DatagramSocket(port);
141                ds.setReuseAddress(true);
142                return true;
143            } catch (IOException e) {
144            } finally {
145                if (ds != null) {
146                    ds.close();
147                }
148
149                if (ss != null) {
150                    try {
151                        ss.close();
152                    } catch (IOException e) {
153                        /* should not be thrown */
154                    }
155                }
156            }
157
158            return false;
159        }
160
161        public Integer getFreePort(int minPort, int maxPort) {
162                Random rand = new Random();
163                int port = -1;
164                do {
165                        port = rand.nextInt((maxPort - minPort) + 1) + minPort;
166                } while (!available(port, minPort, maxPort));
167
168            return port;
169        }
170
171        public Integer getFreePort() {
172                return getFreePort(minPort, maxPort);
173        }
174
175        public void execute() throws MojoExecutionException {           
176                super.execute();
177
178                checkBWEngine();
179                tibcoBWEnginePath = new File(tibcoBWEnginePath.getAbsolutePath());
180
181                initServiceAgent();
182
183                try {
184                        startEngine();
185                } catch (IOException e) {
186                        throw new MojoExecutionException(getServiceFailureMessage(), e);
187                }
188
189                int maxRetry = getMaxRetry();
190                
191                while (!isStarted() && (maxRetry > 0)) {
192                        try {
193                                Thread.sleep(getRetryInterval() * 1000);
194                        } catch (InterruptedException iee) {
195                                iee.printStackTrace(); // FIXME : remove printStackTrace()
196                        }
197                        maxRetry--;
198                }
199
200                if (!isStarted()) {
201                        throw new MojoExecutionException(getServiceTimeoutMessage());
202                }
203
204                getLog().debug("Engine started");
205
206                // now that the Service is started, concrete children can call the ops
207                executeServiceMethods();
208
209                // stopping BusinessWorks engine
210                getLog().info(BWENGINE_STOPPING);
211                stopEngine();
212        }
213
214        private void startEngine() throws IOException, MojoExecutionException {
215                Properties engineProperties = new Properties();
216                engineProperties.setProperty("tibco.clientVar." + getServiceName() + "/HTTP-service-port", getBWEnginePort());
217                engineProperties.setProperty("bw.plugin.jms.recoverOnStartupError", "true");
218                
219//              engineProperties.setProperty("Hawk.Enabled", "true");
220//              engineProperties.setProperty("ServiceAgent.builtinResource.serviceagent.Class", "com.tibco.plugin.brp.BRPServiceAgent");
221//              engineProperties.setProperty("bw.platform.services.retreiveresources.Enabled", "true");
222
223                // création du fichier de propriétés (chargé dans le BWEngine avec l'option "-p")
224                File propertyFile = new File(directory, "bwengine.properties");
225                propertyFile.createNewFile();
226
227                FileOutputStream fos = new FileOutputStream(propertyFile);
228                engineProperties.store(fos, null);
229                fos.close();
230                
231                // merge aliases file into the properties file of the bwengine
232                File aliasesFile = new File(directory, ALIASES_FILE);
233                if (aliasesFile.exists()) {
234                        FileUtils.write(propertyFile, FileUtils.readFileToString(aliasesFile), true);
235                }
236                
237                ArrayList<String> arguments = new ArrayList<String>();
238                arguments.add("-name"); 
239                arguments.add(getServiceName() + " Web Service");
240                arguments.add("-p"); // properties file
241                //arguments.add("-a"); // alias file
242                arguments.add(propertyFile.getAbsolutePath());
243                arguments.add(getProjectToRunPath().getAbsolutePath());
244                getLog().info(BWENGINE_STARTING);
245
246                ArrayList<File> tras = new ArrayList<File>();
247                tras.add(tibcoBWEngineTRAPath);
248                
249                launchTIBCOBinary(tibcoBWEnginePath, tras, arguments, directory, getServiceFailureMessage(), false, false);
250        }
251
252        protected abstract File getProjectToRunPath();
