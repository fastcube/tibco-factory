151        /**
152         * 
153         * @param configuration, a list of "-D" properties
154         * @return the configuration as a List<Element> used by MojoExecutor
155         */
156        private List<Element> generateConfiguration(Properties configuration) {
157                List<Element> configurations = new ArrayList<Element>();
158                                
159                if (configuration != null) {
160                        for (Object key : configuration.keySet()) {
161                                String value = configuration.get(key).toString();
162                                getLog().debug(value);
163                                configurations.add(element(key.toString(), value));
164                        }
165                }
166
167                return configurations;
168        }

174        @SuppressWarnings("deprecation") // this can't be avoided because 'org.apache.maven.plugins:maven-archetype-plugin:2.2:generate' uses this method
175        @Override
176        protected boolean performAction(AbstractProject p) {
177                try {
178                        Properties configuration = new Properties();
179
180                        configuration.put("basedir", p.root + File.separator + p.getRelativePath());
181                        configuration.put("archetypeGroupId", archetypeGroupId);
182                        configuration.put("archetypeArtifactId", archetypeArtifactId);
183
184                        if (archetypeVersion != null) {
185                                configuration.put("archetypeVersion", archetypeVersion);
186                        }
187                        if (archetypeCatalog != null) {
188                                configuration.put("archetypeCatalog", archetypeCatalog);
189                        }
190                        if (archetypeRepository != null) {
191                                configuration.put("archetypeRepository", archetypeRepository);
192                        }
193
194                        configuration.put("goals", archetypeGoals);
195                        configuration.put("interactiveMode", false);
196
197                        archetypeProperties = initProperties(p);
198                        
199                        for (Object s : archetypeProperties.keySet()) {
200                                String key = (String) s;
201                                String value = archetypeProperties.getProperty(key);
202                                
203                                session.getExecutionProperties().put(key, value);                               
204                        }
205                        
206                        executeMojo(
207                        plugin(
208                            groupId("org.apache.maven.plugins"),
209                            artifactId("maven-archetype-plugin"),
210                            version("2.2")
211                        ),
212                        goal("generate"),
213                        configuration(
214                                generateConfiguration(configuration).toArray(new Element[0])
215                        ),
216                        getEnvironment()
217                    );          
218
219                } catch (Exception e) {
220                        getLog().warn("Mavenization failed : " + e.getLocalizedMessage());
221                        return false;
222                }
223
224                return true;
225        }

236        private String getRootProjectPath(AbstractProject p) {
237                String result = "";
238
239                Model parentModel = new Model();
240                parentModel.setGroupId(rootProjectGroupId);
241                parentModel.setArtifactId(rootProjectArtifactId);
242                parentModel.setVersion(rootProjectVersion);
243                parentModel.setPackaging("pom");
244                
245                try {
246                        File parent = getParent(parentModel, new File(p.getOriginalAbsolutePath()).getParentFile(), getLog());
247                        if (parent != null) {
248                                result = parent.getAbsolutePath();
249                        }
250                } catch (Exception e) {
251                }
252                
253                return result;
254        }

269        private Properties initProperties(AbstractProject p) {
270                if (archetypeProperties == null) {
271                        archetypeProperties = new Properties();                 
272                }
273
274                archetypeProperties.put("rootProjectGroupId", rootProjectGroupId);
275                archetypeProperties.put("rootProjectArtifactId", rootProjectArtifactId);
276                archetypeProperties.put("rootProjectVersion", rootProjectVersion);
277
278                String s = AbstractProjectsListMojo.getRelativePath(p.getAbsolutePath(), getRootProjectPath(p), File.separator);
279                String rootRelativePath = "../..";
280                for (int i = StringUtils.countMatches(s, File.separator); i > 0; i--) {
281                        rootRelativePath += "/..";
282                }
283                archetypeProperties.put("rootRelativePath", rootRelativePath);
284                
285                if (isProjlib) {
286                        archetypeProperties.put("projectPackaging", AbstractBWMojo.PROJLIB_TYPE);
287                } else {
288                        archetypeProperties.put("projectPackaging", AbstractBWMojo.BWEAR_TYPE);
289                }
290                
291                archetypeProperties.put("bwProjectName", p.getProjectName());
292                archetypeProperties.put("groupId", projectGroupId);                 
293                archetypeProperties.put("artifactId", p.getArtifactId());
294                if (projectVersion == null || projectVersion.isEmpty()) {
295                        projectVersion = "[inherited]";
296                        archetypeProperties.put("versionIsInherited", "true");
297                }
298                archetypeProperties.put("version", projectVersion);                 
299                if (bwProjectLocation == null) {
300                        bwProjectLocation = "";
301                        archetypeProperties.put("bwProjectLocationIsInherited", "true");
302                }
303                archetypeProperties.put("bwProjectLocation", bwProjectLocation);
304
305                archetypeProperties.put("putPOMInParentDirectory", p.getPutPOMInParentDirectory().toString());
306
307                return archetypeProperties;
308        }
309
310        @Override
311        public void execute() throws MojoExecutionException, MojoFailureException {
312                super.execute();
313                if (!doIt) {
314                        getLog().info(NOTHING_WAS_PERFORMED);
315                } else if (projects.size() > 0) {
316                        int failed = projects.size() - successfullyPerformedAction;
317
318                        if (failed > 1) {
319                                getLog().info(failed + MAVENIZATION_FAILURES);
320                        } else {
321                                getLog().info(failed + MAVENIZATION_FAILURE);
322                        }
323                        if (successfullyPerformedAction > 1) {
324                                getLog().info(successfullyPerformedAction + MAVENIZATION_SUCCESSES);
325                        } else {
326                                getLog().info(successfullyPerformedAction + MAVENIZATION_SUCCESS);
327                        }
328                }
329        }
