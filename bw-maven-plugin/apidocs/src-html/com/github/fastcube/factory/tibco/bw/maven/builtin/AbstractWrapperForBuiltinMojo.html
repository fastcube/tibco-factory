048/**
049 * <p>
050 * This goal is just a wrapper to launch a goal from a builtin Maven plugin
051 * (such as org.apache.maven.plugins:maven-dependency-plugin) with a
052 * <b>configuration bound to the lifecycle but without triggering a parallel
053 * lifecycle</b>. <br />
054 * This configuration will be injected from the Plexus 'components.xml' file.
055 * </p>
056 * 
057 * <p>
058 * <b>NB</b>: The Maven 2 doclet annotations must be used to inject properly the
059 * configuration.
060 * </p>
061 * 
062 * <p>
063 * Duplicate this template code to create a new wrapped goal (all << >> must
064 * be replaced).
065 * </p>
066 * 
067 * <p>
068 * <u>Original goal</u> :
069 * <b>&lt;&lt;groupId>>:&lt;&lt;artifactId>>:&lt;&lt;goal-name>></b>
070 * </p>
071 * 
072 * <p>
073 * Associated Plexus 'components.xml' sections (first section is shared) : 
074 * <pre>
075 * &lt;component>
076 *  &lt;role>org.apache.maven.lifecycle.mapping.LifecycleMapping&lt;/role>
077 *    &lt;role-hint>&lt;&lt;packaging-name>>&lt;/role-hint>
078 *    &lt;implementation>org.apache.maven.lifecycle.mapping.DefaultLifecycleMapping
079 *    &lt;/implementation>
080 *   &lt;configuration>
081 *    &lt;lifecycles>
082 *     &lt;lifecycle>
083 *      &lt;id>default&lt;/id>
084 *      &lt;phases>
085 *       &lt;generate-sources>
086 *        &lt;&lt; io.github.fastcube.factory.tibco.bw.maven:bw-maven-plugin:resolve-bw-dependencies >>
087 *       &lt;/generate-sources>
088 *      &lt;/phases>
089 *     &lt;/lifecycle>
090 *    &lt;/lifecycles>
091 *   &lt;/configuration>
092 *      &lt;/component>
093 *
094 *      &lt;component>
095 *              &lt;role>org.apache.maven.plugin.Mojo&lt;/role>
096 *              &lt;role-hint>&lt;&lt;default-new-goal-name>>&lt;/role-hint>
097 *              &lt;implementation>&lt;&lt;io.github.fastcube.factory.tibco.bw.maven.builtin.AbstractWrapperForBuiltinMojo>>&lt;/implementation>
098 *              &lt;configuration>
099 *                      &lt;groupId>&lt;&lt;org.apache.maven.plugins>>&lt;/groupId>
100 *                      &lt;artifactId>&lt;&lt;maven-dependency-plugin>>&lt;/artifactId>
101 *                      &lt;version>&lt;&lt;2.8>>&lt;/version>
102 *                      &lt;goal>&lt;&lt;resolve>>&lt;/goal>
103 *                      &lt;&lt; &lt;outputFile>${project.build.directory}/resolved&lt;/outputFile> >>
104 *                      &lt;&lt; &lt;includeTypes>projlib,jar&lt;/includeTypes>      >>
105 *              &lt;/configuration>
106 *              &lt;requirements>
107 *                      &lt;requirement>
108 *                              &lt;role>org.apache.maven.plugin.BuildPluginManager&lt;/role>
109 *                              &lt;role-hint />
110 *                              &lt;field-name>pluginManager&lt;/field-name>
111 *                      &lt;/requirement>
112 *              &lt;/requirements>
113 *      &lt;/component>
114 * </pre>
115 * </p>
116 * 
117 * @_goal <<new-goal-name>>
118 * @inheritByDefault true
119 * @requiresProject true
120 * @aggregator true
121 * @requiresDependencyResolution test
122 *
123 * @author Radwane Mabchour
124 * 
125 */
126public abstract class AbstractWrapperForBuiltinMojo<CustomResource extends Resource> extends AbstractMojo {
127        // Mojo configuration
128        /**
129         *  @parameter property="groupId"
130         */
131        protected String groupId;
132        protected abstract String getGroupId();
133
134        /**
135         *  @parameter property="artifactId"
136         */
137        protected String artifactId;
138        protected abstract String getArtifactId();
139
140        /**
141         *  @parameter property="version"
142         */
143        protected String version;
144        protected abstract String getVersion();
145
146        /**
147         *  @parameter property="goal}
148         */
149        protected String goal;
150        protected abstract String getGoal();
151
152        // Environment configuration
153        /**
154         * The project currently being build.
155         *
156         * @parameter property="project"
157         * @required
158         * @readonly
159         */
160        protected MavenProject project;
161        protected abstract MavenProject getProject();
162
163        /**
164         * The current Maven session.
165         *
166         * @parameter property="session"
167         * @required
168         * @readonly
169         */
170        protected MavenSession session;
171        protected abstract MavenSession getSession();
172
173        /**
174         * The Build Plugin Manager (this one is Java5 annotation style).
175         */
176        @Component (role = BuildPluginManager.class)
177        protected BuildPluginManager pluginManager;
178        protected abstract BuildPluginManager getPluginManager();
179
180        // Configuration
181    /**
182     * The actual Mojo configuration found in the Plexus 'components.xml' file.
183     * <!--
184         *              <configuration>
185         *                      ...
186         *              <<      <properties>
187         *                              <property>
188         *                                      <name><<outputFile>></name>
189         *                                      <value><<${project.build.directory}/resolved>></value>
190         *                              </property>
191         *                              <property>
192         *                                      <name><<includeTypes>></name>
193         *                                      <value><<projlib,jar>></value>
194         *                              </property>
195         *                      </properties> >>
196         *              </configuration>
197     * -->
198     * @parameter
199     */
200        protected Properties configuration;
201    protected abstract Properties getConfiguration();
202        
203    /**
204    * Optional resources parameter do define includes/excludes filesets
205    * 
206    * @parameter
207    */
208        protected Properties resources;
209    protected abstract List<CustomResource> getResources();
210
211        // Execution
212        /**
213         * @return the Execution Environment for the actual Mojo being called
214         */
215        protected ExecutionEnvironment getEnvironment() {
216                return executionEnvironment(getProject(), getSession(), getPluginManager());
217        }
218
219        /**
220         * Children classes can use a descendant of {@link Resource} (because of the
221         * <? extends Resource> declaration).
222         * If so, the children classes are responsible for coding the additional 
223         * mappings from Java objects to {@link Element}.
224         * @param resource 
225         *
226         *@return null, children classes need to override this method to parse
227         * their custom elements
228         */
229        protected List<Element> generateCustomElements(CustomResource resource) {
230                return null;
231        }
232
233        private List<Element> generateResources() {
234                List<CustomResource> resources = getResources();
235                List<Element> resourcesElement = new ArrayList<Element>();
236
237                if (resources != null) {
238                        getLog().debug(resources.toString());
239                        
240                        for (CustomResource resource : resources) {
241                                List<Element> children = new ArrayList<Element>();
242                                children.add(element("directory", resource.getDirectory()));
243                                children.add(element("filtering", resource.getFiltering()));
244                                children.add(element("targetPath", resource.getTargetPath()));
245                                
246                                List<Element> excludes = new ArrayList<Element>();
247                                for (String exclude : resource.getExcludes()) {
248                                        excludes.add(element("exclude", exclude));
249                                }
250                                children.add(element("excludes", excludes.toArray(new Element[0])));
251
252                                List<Element> includes = new ArrayList<Element>();
253                                for (String include : resource.getIncludes()) {
254                                        excludes.add(element("include", include));
255                                }
256                                children.add(element("includes", includes.toArray(new Element[0])));
257
258                                List<Element> customElements = generateCustomElements(resource);
259                                if (customElements != null) {
260                                        for (Element element : customElements) {
261                                                children.add(element);
262                                        }
263                                }
264
265                                resourcesElement.add(element("resource", children.toArray(new Element[0])));
266                        }                       
267                }
268
269                return resourcesElement;
270        }
271
272        /**
273         * 
274         * @param configuration, the configuration found in Plexus 'components.xml'
275         * file
276         * @return the configuration as a List<Element> used by MojoExecutor
277         * @throws IOException 
278         */
279        private List<Element> generateConfiguration(Properties configuration) {
280                ArrayList<Element> configurations = new ArrayList<Element>();
281
282                if (configuration != null) {
283                        for (Object key : configuration.keySet()) {
284                                String value = configuration.get(key).toString();
285                                getLog().debug(value);
286                                if ("properties".equals(key) && value != null) {
287                                        getLog().debug("key='properties'");
288                                        Pattern p = Pattern.compile("^\\$\\{(.*)\\}$");
289                                        Matcher matcherElement = p.matcher(value);
290                                        if (matcherElement.matches()) {
291                                                getLog().debug("matching properties to expand");
292                                                value = matcherElement.group(1);
293                                                String replacedValue = this.getProject().getProperties().getProperty(value);
294                                                if (replacedValue != null && !replacedValue.isEmpty()) {
295                                                        value = replacedValue;
296                                                        getLog().debug("properties_value: " + value);
297                                                        Properties properties = new Properties();
298                                                        try {
299                                                                properties.load(new StringReader(value));
300                                                                getLog().debug(properties.toString());
301                                                        } catch (IOException e) {
302                                                                continue;
303                                                        }
304
305                                                        List<Element> children = new ArrayList<Element>();
306                                                        for (Object o : properties.keySet()) {
307                                                                String childKey = (String) o;
308                                                                String childValue = properties.getProperty(childKey);
309                                                                children.add(element(childKey, childValue));
310                                                        }
311
312                                                        configurations.add(element(key.toString(), children.toArray(new Element[0])));
313                                                }
314                                        }
315
316                                } else {
317                                        configurations.add(element(key.toString(), value));
318                                }
319                        }
320                }
321
322                List<Element> resources = generateResources();
323                if (resources != null && !resources.isEmpty()) {
324                        configurations.add(element("resources", resources.toArray(new Element[0])));
325                }
326
327                return configurations;
328        }
329
330        /**
331         * This method will call the actual builtin goal from the Mojo specified in
332         * Plexus 'components.xml'.<br/>
333         * 
334         * The {@link MojoExecutor} is used to perform the call natively.
335         */
336        public void execute() throws MojoExecutionException {
337                executeMojo(
338                plugin(
339                    groupId(getGroupId()),
340                    artifactId(getArtifactId()),
341                    version(getVersion())
342                ),
343                goal(getGoal()),
344                configuration(
345                        generateConfiguration(getConfiguration()).toArray(new Element[0])
346                ),
347                getEnvironment()
348            );          
349        }
