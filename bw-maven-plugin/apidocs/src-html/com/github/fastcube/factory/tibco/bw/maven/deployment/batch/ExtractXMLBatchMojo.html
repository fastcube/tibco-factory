017package com.github.fastcube.factory.tibco.bw.maven.deployment.batch;
018
019import java.util.ArrayList;
020
021import org.apache.maven.plugin.MojoExecutionException;
022import org.apache.maven.plugins.annotations.LifecyclePhase;
023import org.apache.maven.plugins.annotations.Mojo;
024import org.apache.maven.plugins.annotations.Parameter;
025
026/**
027 * <p>
028 * This goal performs an export of a TIBCO domain to a directory.
029 * </p>
030 * <p>
031 * The result can be used by batch goals such as
032 * <a href="./batch-delete-bw-mojo.html">bw:batch-delete-bw</a>,
033 * <a href="./batch-deploy-bw-mojo.html">bw:batch-deploy-bw</a>,
034 * <a href="./batch-start-bw-mojo.html">bw:batch-start-bw</a>,
035 * <a href="./batch-stop-bw-mojo.html">bw:batch-stop-bw</a> or
036 * <a href="./batch-undeploy-bw-mojo.html">bw:batch-undeploy-bw</a>.
037 * </p>
038 * @author Radwane Mabchour
039 *
040 */
041@Mojo(name="batch-extract-xml-bw", defaultPhase=LifecyclePhase.PREPARE_PACKAGE, requiresProject = false)
042public class ExtractXMLBatchMojo extends AbstractBWBatchDeployMojo {
043
044        protected final static String EXTRACTING_CONFIG_BATCH = "Extracting configuration from domain (batch)...";
045        protected final static String EXTRACTING_CONFIG_SUCCESS = "Successfully extracted configuration from domain (batch) to ";
046        protected final static String EXTRACTING_CONFIG_FAILED = "The extraction (batch) of the XML Deployment Descriptor files from the deployed application file has failed.";
047
048        /**
049         * <p>
050         * Whether to extract in template format or not
051         * (<i>-template</i> switch from AppManage).
052         * </p>
053         */
054        @Parameter ( property = deployBatchTemplate, required = true , defaultValue = "false" )
055        protected boolean batchTemplate;
056
057        /**
058         * <p>
059         * Whether to perform a minimal export or not
060         * (<i>-min</i> switch from AppManage).
061         * </p>
062         */
063        @Parameter ( property = deployBatchMin, required = true , defaultValue = "false" )
064        protected boolean batchMin;
065
066        /**
067         * <p>
068         * Whether to perform a maximal export or not
069         * (<i>-max</i> switch from AppManage).
070         * </p>
071         */
072        @Parameter ( property = deployBatchMax, required = true , defaultValue = "false" )
073        protected boolean batchMax;
074
075        /**
076         * <p>
077         * Whether to extract EAR or not
078         * (<i>-noear</i> switch from AppManage).
079         * </p>
080         */
081        @Parameter ( property = deployBatchNoEAR, required = true , defaultValue = "false" )
082        protected boolean batchNoEAR;
083
084        /**
085         * <p>
086         * Whether to export deployed configuration
087         * (<i>-exportDeployed</i> switch from AppManage).
088         * </p>
089         */
090        @Parameter ( property = deployBatchExportDeployed, required = true , defaultValue = "false" )
091        protected boolean batchExportDeployed;
092
093        @Override
094        public String getInitMessage() {
095                super.getInitMessage();
096                return EXTRACTING_CONFIG_BATCH;
097        }
098
099        @Override
100        public String getFailureMessage() {
101                return EXTRACTING_CONFIG_FAILED;
102        }
103
104        @Override
105        public void postAction() throws MojoExecutionException {
106                getLog().info(EXTRACTING_CONFIG_SUCCESS + "'" + dir.getAbsolutePath() + "'.");
107        }
108
109        @Override
110        public ArrayList<String> arguments() {
111                ArrayList<String> arguments = super.commonArguments();
112
113                arguments.add("-batchExport");
114                if (batchTemplate) {
115                        arguments.add("-template");
116                }
117                if (batchMin) {
118                        arguments.add("-min");
119                }
120                if (batchMax) {
121                        arguments.add("-max");
122                }
123                if (batchNoEAR) {
124                        arguments.add("-noear");
125                }
126                if (batchExportDeployed) {
127                        arguments.add("-exportDeployed");
128                }
129
130                return arguments;
131        }
132
133}
