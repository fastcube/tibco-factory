043/**
044 * 
045 * <p>
046 * This goal starts a TIBCO BusinessWorks application deployed on a TIBCO domain.
047 * </p>
048 * 
049 * @author Radwane Mabchour
050 *
051 */
052@Mojo( name="start-bw",
053defaultPhase=LifecyclePhase.DEPLOY ) // FIXME: should be deployEAR
054public class StartEARMojo extends AbstractBWDeployMojo {
055
056        protected final static String STARTING_INSTANCES_FAILED = "Some instances failed to be started.";
057        protected final static String STARTING_INSTANCES = "Starting instances of the application...";
058        protected final static String WAITING_FOR_INSTANCES = "Waiting for instances to be started...";
059        protected final static String SUBSCRIBING_TO_HAWK_METHODS = "Subscribing to Hawk methods:";
060        protected final static String ALL_INSTANCES_STARTED = "All instances successfully started.";
061        protected final static String SOME_INSTANCES_NOT_STARTED = "Some instances failed to be started (or timeout was reached).";
062
063        /**
064         * Whether to wait for instances to be started or not.
065         * <br />
066         * The waiting process is performed with Hawk methods subscriptions.
067         * <br /><br />
068         * <b>NB</b>: it is mandatory to configure Hawk properly.
069         */
070        @Parameter (property="waitForRunningInstances", defaultValue = "false")
071        private Boolean waitForRunningInstances;
072
073        /**
074         * Whether to fail the Maven build when timeout is reached or not.
075         */
076        @Parameter (property="failWhenTimeoutReached", defaultValue = "false")
077        private Boolean failWhenTimeoutReached;
078        
079        @Parameter (property="bw.start.running.statuses.number", defaultValue = "3")
080        private int numberOfRunningStatuses;
081        private MicroAgent microAgent;
082
083        private static class RunningInstanceSubscriptionHandler extends DefaultSubscriptionHandler {
084                private Log logger;
085                private MethodSubscriber methodSubscriber;
086
087                private enum StatusHistory {
088                        STOPPED, INITIALIZING, RUNNING
089                }
090                private Map<Integer, StatusHistory> statusesHistory;
091                private int numberOfRunningStatuses;
092
093                public RunningInstanceSubscriptionHandler(Log log, MethodSubscriber methodSubscriber, int numberOfRunningStatuses) {
094                        this.logger = log;
095                        this.methodSubscriber = methodSubscriber;
096                        this.statusesHistory = new LinkedHashMap<Integer, StatusHistory>();
097                        if (numberOfRunningStatuses <= 0) {
098                                numberOfRunningStatuses = 1;
099                        }
100                        this.numberOfRunningStatuses = numberOfRunningStatuses;
101                }
102
103                protected Log getLog() {
104                        return logger;
105                }
106
107                @Override
108                public Object getResult() {
109                        return result;
110                }
111
112                @Override
113                public void onSubscribe() {
114                        String instance =
115                        methodSubscriber.getArguments().get(0).getValue() +
116                        "-" +
117                        methodSubscriber.getArguments().get(1).getValue();
118
119                        Integer timeout = methodSubscriber.getInterval() * methodSubscriber.getNumberOfRetry() / 1000;
120                        getLog().info("Subscription to the Hawk method started for instance '" + instance + "' (timeout is set to " + timeout + (timeout > 1 ? " seconds" : " second") + ").");
121                }
122
123                @Override
124                public void onData(Subscription s, MicroAgentData mad) {
125                        super.onData(s, mad);
126                        String data = formatData(mad.getData());
127                        getLog().debug(data);
128
129                        if (data != null && data.contains("STOPPED")) {
130                                statusesHistory.put(statusesHistory.size(), StatusHistory.STOPPED);
131                        } else  if (data != null && data.contains("INITIALIZING")) {
132                                statusesHistory.put(statusesHistory.size(), StatusHistory.INITIALIZING);
133                        } else  if (data != null && data.contains("RUNNING")) {
134                                statusesHistory.put(statusesHistory.size(), StatusHistory.RUNNING);
135                        }
136
137                        getLog().debug(statusesHistory.toString());
138                        if (failedToStart()) {
139                                s.cancel();
140                                result = "STOPPED";
141                                decreaseReferenceCount();
142                        } else if (isStarted()) {
143                                s.cancel();
144                                result = "RUNNING";
145                                decreaseReferenceCount();
146                                getLog().debug("A subscription was canceled");
147                                getLog().debug(numberOfSubscriptions.toString());
148                                getLog().debug(RunningInstanceSubscriptionHandler.numberOfSubscriptions.toString());
149                        }
150                }
151
152                private Integer getLastKeyForValue(Map<Integer, StatusHistory> map, StatusHistory value) {
153                        Integer result = -1;
154
155                        for (Integer key : map.keySet()) {
156                                if (value.equals(map.get(key))) {
157                                        result = key;
158                                }
159                                
160                        }
161                        return result;
162                }
163
164                private boolean failedToStart() {
165                        if (statusesHistory.containsValue(StatusHistory.RUNNING) &&
166                                statusesHistory.containsValue(StatusHistory.STOPPED)) {
167                                        Integer stoppedIndex = getLastKeyForValue(statusesHistory, StatusHistory.STOPPED);
168                                        getLog().debug(stoppedIndex.toString());
169                                        Integer runningIndex = getLastKeyForValue(statusesHistory, StatusHistory.RUNNING);
170                                        getLog().debug(runningIndex.toString());
171                                        return stoppedIndex > runningIndex;
172                                }
173                        return false;
174                }
175
176                private boolean isStarted() {
177                        return Collections.frequency(statusesHistory.values(), StatusHistory.RUNNING) > numberOfRunningStatuses;
178                }
179
180                @Override
181                public void decreaseReferenceCount() {
182                        super.decreaseReferenceCount();
183                        getLog().debug(numberOfSubscriptions.toString());
184                }
185
186                @Override
187                public void increaseReferenceCount() {
188                        super.increaseReferenceCount();
189                        getLog().debug(numberOfSubscriptions.toString());
190                }
191                
192                @Override
193                public void onError(Subscription s, MicroAgentException e) {
194                        getLog().debug("A subscription caused an error");
195                }
196
197                @Override
198                public void onErrorCleared(Subscription s) {
199                        getLog().debug("A subscription cleared an error");
200                }
201
202                @Override
203                public void onTermination(Subscription s, MicroAgentException e) {
204                        getLog().debug("A subscription terminated");
205                }
206
207        }
208        
209        /**
210         * This methods subscribe to the Hawk method
211         * "COM.TIBCO.admin.TRA:getComponentInstanceStatus" for all
212         * {@code instances} being started.
213         * 
214         * @param instances
215         * @return a list of {@link MethodSubscriber} objects
216         * @throws ConsoleInitializationException
217         * @throws MicroAgentException
218         */
219        private List<MethodSubscriber> monitorInstances(List<ImmutablePair<String, String>> instances) throws ConsoleInitializationException, MicroAgentException {
220                List<MethodSubscriber> result = new ArrayList<MethodSubscriber>();
221
222        microAgent = new MicroAgent(hawkDomain, hawkRvService, hawkRvNetwork, hawkRvDaemon, "COM.TIBCO.admin.TRA");
223
224        getLog().info("");
225                getLog().info(SUBSCRIBING_TO_HAWK_METHODS);
226
227        for (ImmutablePair<String, String> instance : instances) {
228                MethodSubscriber ms = new MethodSubscriber(microAgent);
229                ms.setMethodName("getComponentInstanceStatus");
230                ms.setArguments(instance.left, instance.right);
231                ms.setInterval(hawkSubscribeInterval);
232                ms.setNumberOfRetry(hawkSubscribeNumberOfRetry);
233                ms.subscribe(new RunningInstanceSubscriptionHandler(getLog(), ms, numberOfRunningStatuses));
234
235                result.add(ms);
236                }
237
238                return result;
239        }
240
241        private boolean doWaitForRunningInstances(List<ImmutablePair<String, String>> instances) throws ConsoleInitializationException, MicroAgentException {
242                List<MethodSubscriber> result = monitorInstances(instances); // start monitoring of "RUNNING" statuses for instances
243
244                getLog().info("");
245                getLog().info(WAITING_FOR_INSTANCES);
246
247                while (RunningInstanceSubscriptionHandler.numberOfSubscriptions > 0) {
248                        getLog().debug(RunningInstanceSubscriptionHandler.numberOfSubscriptions.toString());
249                        try {
250                                Thread.sleep(1000);
251                        } catch (InterruptedException e) {
252                                e.printStackTrace();
253                        }
254                        /*
255                         *  RunningInstanceSubscriptionHandler.numberOfSubscriptions will
256                         * always come down to 0. In worst case after timeout. 
257                         */
258                }
259
260                microAgent.shutdown();
261
262                // check that all instances are in "RUNNING" state
263                for (MethodSubscriber ms : result) {
264                        if (ms.getResult() == null || !ms.getResult().equals("RUNNING")) {
265                                return false;
266                        }
267                }
268                return true;
269        }
270
271        @Override
272        public String getInitMessage() {
273                return STARTING_INSTANCES;
274        }
275
276        @Override
277        public String getFailureMessage() {
278                return STARTING_INSTANCES_FAILED;
279        }
280
281        @Override
282        public ArrayList<String> arguments() {
283                ArrayList<String> arguments = super.commonArguments();
284                arguments.add("-start");
285
286                return arguments;
287        }
288
289        @Override
290        public void postAction() throws MojoExecutionException {
291                try {
292                        if (waitForRunningInstances) {
293                                if (!initHawk(false)) {
294                                        return;
295                                }
296
297                                List<ImmutablePair<String, String>> instances = new ArrayList<ImmutablePair<String,String>>();
298
299                                ApplicationManagement application = new ApplicationManagement(deploymentDescriptorFinal);
300                                String applicationName = application.getName();
301                                List<String> servicesNames = application.getInstancesNames(true);
302
303                                for (String serviceName : servicesNames) {
304                                        instances.add(new ImmutablePair<String, String>(applicationName, serviceName));
305                                }
306
307                                if (doWaitForRunningInstances(instances)) {
308                                        getLog().info("");
309                                        getLog().info(ALL_INSTANCES_STARTED);
310                                } else {
311                                        getLog().info("");
312                                        if (failWhenTimeoutReached) {
313                                                throw new MojoExecutionException(SOME_INSTANCES_NOT_STARTED);
314                                        } else {
315                                                getLog().info(SOME_INSTANCES_NOT_STARTED);
316                                        }
317                                }
318                        }
319                } catch (JAXBException e) {
320                        throw new MojoExecutionException(SOME_INSTANCES_NOT_STARTED);
321                } catch (ConsoleInitializationException e) {
322                        throw new MojoExecutionException(SOME_INSTANCES_NOT_STARTED);
323                } catch (MicroAgentException e) {
324                        throw new MojoExecutionException(SOME_INSTANCES_NOT_STARTED);
325                }
326
327        }
