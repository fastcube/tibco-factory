050/**
051 * <p>
052 * This goal includes the JAR dependencies of the runtime scope inside the
053 * "lib.zip" of the TIBCO BusinessWorks EAR.<br/>
054 * It allows to load seamlessly the transitive dependencies of JARs of the
055 * TIBCO BusinessWorks project and avoid the ClassNotFoundException issues when
056 * the application is deployed on a TIBCO domain.
057 * </p>
058 * <p>
059 * This step can be ignored by setting <i>includeTransitiveJARsInEAR</i> to
060 * <i>false</i>.
061 * </p>
062 * 
063 * @see UpdateAliasesLibsMojo
064 * @author Radwane Mabchour
065 * 
066 */
067@Mojo( name="include-dependencies-in-bw-ear",
068defaultPhase=LifecyclePhase.COMPILE )
069public class IncludeDependenciesInEARMojo extends AbstractBWArtifactMojo {
070
071        /**
072         * Whether to add JARs files inside EAR.
073         */
074        @Parameter (property="includeTransitiveJARsInEAR", defaultValue="true")
075        public Boolean includeTransitiveJARsInEAR;
076
077        /**
078         * Whether to rename JARs files inside EAR without their version.
079         */
080        @Parameter (property="removeVersionFromFileNames", defaultValue="false")
081        public Boolean removeVersionFromFileNames;
082        
083        @Override
084        protected String getArtifactFileExtension() {
085                return BWEAR_EXTENSION;
086        }
087
088        private DefaultTrueZip truezip;
089        
090        /**
091         * <p>
092         * This methods copies the transitive JAR dependencies of the project inside
093         * the "WEB-INF/lib" folder of the "lib.zip" subarchive of the TIBCO
094         * BusinessWorks EAR archive.
095         * </p>
096         * 
097         * @param ear, the TIBCO BusinessWorks EAR archive file
098         * @throws IOException 
099         * @throws JDOMException 
100         * @throws JaxenException 
101         */
102        private void copyRuntimeJARsInEAR(File ear) throws IOException, JDOMException {
103                Fileset fileSet = new Fileset();
104                fileSet.setDirectory(buildLibDirectory.getAbsolutePath());
105
106                for (Dependency dependency : this.getJarDependencies()) {
107                        String jarName = getJarName(dependency, false);
108                        
109                        fileSet.addInclude(jarName); // using jarName because files are all in buildLibDirectory
110                }
111                
112                String ouptutDirectory = ear.getAbsolutePath() + File.separator + "lib.zip" + File.separator + "WEB-INF" + File.separator + "lib";
113        fileSet.setOutputDirectory(ouptutDirectory);        
114        
115        if (fileSet.getIncludes() != null && !fileSet.getIncludes().isEmpty()) {
116                truezip.copy(fileSet);
117        }
118                truezip.sync();
119                
120                if (removeVersionFromFileNames) {
121                        removeVersionFromFileNames(ouptutDirectory, ear);
122
123                        truezip.sync();
124                }
125        }
126
127        private void removeVersionFromFileNames(String ouptutDirectory, File ear) throws IOException, JDOMException {
128                for (Dependency dependency : this.getJarDependencies()) {
129                        Pattern p = Pattern.compile("(.*)-" + dependency.getVersion() + JAR_EXTENSION);
130
131                        String includeOrigin = getJarName(dependency, false);
132                        String includeDestination;
133
134                        Matcher m = p.matcher(includeOrigin);
135                        if (m.matches()) {
136                                includeDestination = m.group(1)+JAR_EXTENSION;
137                                
138                                truezip.moveFile(new TFile(ouptutDirectory + File.separator + includeOrigin), new TFile(ouptutDirectory + File.separator + includeDestination));
139                                
140                                updateAlias(includeOrigin, includeDestination, ear);
141                        }
142                }
143
144                truezip.sync();
145        }
146
147        private void updateAlias(String includeOrigin, String includeDestination, File ear) throws JDOMException, IOException, JDOMException {
148                TFile xmlTIBCO = new TFile(ear.getAbsolutePath() + File.separator + "TIBCO.xml");
149                String tempPath = ear.getParentFile().getAbsolutePath() + File.separator + "TIBCO.xml";
150                TFile xmlTIBCOTemp = new TFile(tempPath);
151
152                truezip.copyFile(xmlTIBCO, xmlTIBCOTemp);
153
154                File xmlTIBCOFile = new File(tempPath);
155
156                SAXBuilder sxb = new SAXBuilder();
157                Document document = sxb.build(xmlTIBCOFile);
158
159                XPath xpa = XPath.newInstance("//dd:NameValuePairs/dd:NameValuePair[starts-with(dd:name, 'tibco.alias') and dd:value='" + includeOrigin + "']/dd:value");
160                xpa.addNamespace("dd", "http://www.tibco.com/xmlns/dd");
161
162                Element singleNode = (Element) xpa.selectSingleNode(document);
163                if (singleNode != null) {
164                        singleNode.setText(includeDestination);
165                        XMLOutputter xmlOutput = new XMLOutputter();
166                        xmlOutput.setFormat(Format.getPrettyFormat().setIndent("    "));
167                        xmlOutput.output(document, new FileWriter(xmlTIBCOFile));
168
169                        truezip.copyFile(xmlTIBCOTemp, xmlTIBCO);
170                }
171
172                updateAliasInPARs(includeOrigin, includeDestination, ear);
173        }
174
175        private void updateAliasInPARs(String includeOrigin, String includeDestination, File ear) throws IOException, JDOMException {
176                TrueZipFileSet pars = new TrueZipFileSet();
177                pars.setDirectory(ear.getAbsolutePath());
178                pars.addInclude("*.par");
179                List<TFile> parsXML = truezip.list(pars);
180                for (TFile parXML : parsXML) {
181                        TFile xmlTIBCO = new TFile(parXML, "TIBCO.xml");
182
183                        String tempPath = ear.getParentFile().getAbsolutePath() + File.separator + "TIBCO.xml";
184                        TFile xmlTIBCOTemp = new TFile(tempPath);
185
186                        truezip.copyFile(xmlTIBCO, xmlTIBCOTemp);
187
188                        File xmlTIBCOFile = new File(tempPath);
189
190                        SAXBuilder sxb = new SAXBuilder();
191                        Document document = sxb.build(xmlTIBCOFile);
192
193                        XPath xpa = XPath.newInstance("//dd:NameValuePairs/dd:NameValuePair[dd:name='EXTERNAL_JAR_DEPENDENCY']/dd:value");
194                        xpa.addNamespace("dd", "http://www.tibco.com/xmlns/dd");
195
196                        Element singleNode = (Element) xpa.selectSingleNode(document);
197                        if (singleNode != null) {
198                                String value = singleNode.getText().replace(includeOrigin, includeDestination);
199                                singleNode.setText(value);
200                                XMLOutputter xmlOutput = new XMLOutputter();
201                                xmlOutput.setFormat(Format.getPrettyFormat().setIndent("    "));
202                                xmlOutput.output(document, new FileWriter(xmlTIBCOFile));
203
204                                truezip.copyFile(xmlTIBCOTemp, xmlTIBCO);
205                        }
206                }
207        }
208
209        public void execute() throws MojoExecutionException {
210        if (skipCompile || skipEARCompile) {
211                getLog().info(SKIPPING);
212                return;
213        }
214
215                if (isCurrentGoal("bw:launch-designer") || !includeTransitiveJARsInEAR) {
216                        return; // ignore
217                }
218
219                super.execute();
220
221                File ear = getProject().getArtifact().getFile(); // EAR generated by "compile-bw-ear" goal
222                if (ear == null) {
223                        ear = getOutputFile();
224                }
225                getLog().debug("Using EAR : " + ear.getAbsolutePath());
226
227                TConfig.get().setArchiveDetector( new TArchiveDetector( TArchiveDetector.NULL, new Object[][] {
228                                { "zip|kar|par|ear", new ZipDriver( IOPoolLocator.SINGLETON ) },
229                } ) );
230
231                truezip = new DefaultTrueZip();
232
233                try {
234                        this.copyRuntimeJARsInEAR(ear);
235                } catch (IOException e) {
236                        throw new MojoExecutionException(e.getMessage(), e);
237                } catch (JDOMException e) {
238                        throw new MojoExecutionException(e.getMessage(), e);
239                }
240        }
