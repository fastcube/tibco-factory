017package com.github.fastcube.factory.tibco.bw.maven.packaging;
018
019import java.io.File;
020import java.io.IOException;
021import java.util.ArrayList;
022
023import org.apache.maven.plugin.MojoExecutionException;
024import org.apache.maven.plugins.annotations.LifecyclePhase;
025import org.apache.maven.plugins.annotations.Mojo;
026
027/**
028 * <p>
029 * This goal extracts the XML Deployment Descriptor from a TIBCO BusinessWorks
030 * EAR.
031 * </p>
032 *
033 * @author Radwane Mabchour
034 *
035 */
036@Mojo( name="extract-xml-from-ear",
037defaultPhase=LifecyclePhase.PREPARE_PACKAGE )
038public class ExtractXMLFromEARMojo extends AbstractPackagingMojo {
039
040        protected final static String CREATE_XML_FROM_EAR_SUCCESS = "Successfully extracted the XML Deployment Descriptor to ";
041        protected final static String CREATE_XML_FROM_EAR_FAILED = "The extraction of the XML Deployment Descriptor file from the EAR file has failed.";
042
043        @Override
044        protected String getArtifactFileExtension() {
045                return XML_EXTENSION;
046        }
047
048        private void createXML() throws MojoExecutionException, IOException {
049                checkAppManage();
050
051                String earPath = getProject().getArtifact().getFile().getPath();
052                String xmlOutputFile = deploymentDescriptor.getPath(); 
053                
054                ArrayList<String> arguments = new ArrayList<String>();          
055                arguments.add("-export");
056                arguments.add("-max");
057                arguments.add("-ear");
058                arguments.add(earPath); 
059                arguments.add("-out");
060                arguments.add(xmlOutputFile); 
061
062                ArrayList<File> tras = new ArrayList<File>();
063                tras.add(tibcoAppManageTRAPath);
064
065                launchTIBCOBinary(tibcoAppManagePath, tras, arguments, directory, CREATE_XML_FROM_EAR_FAILED);
066        }
067
068        public void execute() throws MojoExecutionException {
069                if (super.skip()) {
070                        return;
071                }
072
073                try {
074                        createXML();
075                        getLog().info(CREATE_XML_FROM_EAR_SUCCESS + " '" + deploymentDescriptor + "'");
076                } catch (IOException e) {
077                        throw new MojoExecutionException(CREATE_XML_FROM_EAR_FAILED, e);
078                }
079        }
080
081}
