019import static org.apache.commons.io.FileUtils.copyFile;
020
021import java.io.File;
022import java.io.FileInputStream;
023import java.io.FileOutputStream;
024import java.io.IOException;
025import java.util.ArrayList;
026import java.util.Properties;

028import org.apache.maven.plugin.MojoExecutionException;
029import org.apache.maven.plugins.annotations.Execute;
030import org.apache.maven.plugins.annotations.LifecyclePhase;

032import org.apache.maven.plugins.annotations.Parameter;

037/**
038 * <p>
039 * This goal launches a TIBCO Designer with exactly the same environment that
040 * will be used to compile the TIBCO BusinessWorks EAR from the TIBCO
041 * BusinessWorks project sources (with the
042 * <a href="./compile-bw-ear-mojo.html">compile-bw-ear</a> goal in the
043 * <a href="./bwmaven-lifecycles.html#EAR">compile phase</a>).
044 * </p>
045 * <p>
046 * This is made possible by inheriting from {@link AbstractBWMojo} class.
047 * </p>
048 * 
049 * @author Radwane Mabchour
050 * 
051 */
052@Mojo( name="launch-designer", defaultPhase = LifecyclePhase.PROCESS_TEST_RESOURCES )
053@Execute ( phase=LifecyclePhase.PROCESS_TEST_RESOURCES )
054public class LaunchDesignerMojo extends AbstractBWMojo {
055
056        protected final static String LAUNCH_DESIGNER_FAILED = "The launch of the TIBCO Designer has failed.";
057        protected final static String LAUNCHING_DESIGNER = "Launching TIBCO Designer...";
058        protected final static String PROJECT_LOCATION = "Project location : ";
059
060        /**
061         * If true, the project copied in "target/src" is used by TIBCO Designer.
062         */
063        @Parameter(property="bw.project.use.copy", required=false, defaultValue="false")
064        protected Boolean useBuildSrcDirectory;
065        
066        /**
067         * If true, ".aliaslib" can use parameters such as ${project.version}.
068         */
069        @Parameter(property="bw.project.adapt.aliases", required=false, defaultValue="true")
070        protected Boolean adaptAliases;
071
072        /**
073         * This method copies the '.designtimelibs' of the "target/src" directory
074         * to the actual project's source folder
075         * 
076         * @throws IOException
077         */
078        private void copyDesignTimeLibs() throws IOException {
079                copyFile(new File(testSrcDirectory + "/" + DTL_FILE_NAME),
080                                 new File(projectDirectory + "/" + DTL_FILE_NAME));
081        }
082
083        /**
084         * 
085         * 
086         * @throws MojoExecutionException
087         * @throws IOException
088         */
089        private void launchDesigner() throws MojoExecutionException, IOException {
090                checkDesigner();
091
092                copyDesignTimeLibs();
093                
094                ArrayList<String> arguments = new ArrayList<String>();
095                if (useBuildSrcDirectory) {
096                        arguments.add(buildSrcDirectory.getAbsolutePath()); // using the one in "target/src"
097                        getLog().info(PROJECT_LOCATION + buildSrcDirectory.getAbsolutePath());
098                } else {                        
099                        arguments.add(projectDirectory.getAbsolutePath()); // real BW project path (not the one in "target/src")
100                        getLog().info(PROJECT_LOCATION + projectDirectory.getAbsolutePath());
101                }
102
103                getLog().info(LAUNCHING_DESIGNER);
104                
105                ArrayList<File> tras = new ArrayList<File>();
106                tras.add(tibcoDesignerTRAPath);
107
108                launchTIBCOBinary(tibcoDesignerPath, tras, arguments, directory, LAUNCH_DESIGNER_FAILED, true, false);
109        }
110
111        public void execute() throws MojoExecutionException {
112                enableTestScope();
113
114                super.execute();
115
116                if (adaptAliases) {
117                        try {
118                                updateAliasesFile();
119                        } catch (IOException e) {
120                                throw new MojoExecutionException(e.getLocalizedMessage(), e);
121                        }
122                }
123
124                try {
125                        launchDesigner();
126                } catch (IOException e) {
127                        throw new MojoExecutionException(LAUNCH_DESIGNER_FAILED, e);
128                }
129        }
130
131        private void updateAliasesFile() throws IOException {
132                File aliasesFile = getAliasesFile();
133                File designer5Prefs = getDesigner5Prefs();
134
135                Properties prefs = new SortedProperties();
136                FileInputStream fisPrefs = new FileInputStream(designer5Prefs);
137                prefs.load(fisPrefs);
138                fisPrefs.close();
139                
140                Integer maxFileAliasPref = 0;
141                for (Object k : prefs.keySet()) {
142                        String key = (String) k;
143
144                        if (key.startsWith(FILE_ALIAS_PREFIX)) {
145                                maxFileAliasPref++;
146                        }
147                }
148
149                Properties aliases = new Properties();
150                FileInputStream fis = new FileInputStream(aliasesFile);
151                aliases.load(fis);
152                fis.close();
153
154                String projectVersion = getProject().getVersion();
155                Properties duplicates = new Properties();
156
157                for (Object k : aliases.keySet()) {
158                        String key = (String) k;
159                        String value = aliases.getProperty(key);
160                        if (key.contains(projectVersion) && key.endsWith(":jar")) {
161                                getLog().debug(key);
162                                key = key.replace(projectVersion, "${project.version}");
163                                duplicates.put(key, value);
164                        }
165                }
166                
167                if (!duplicates.isEmpty()) {
168                        for (Object k : duplicates.keySet()) {
169                                String key = (String) k;
170                                String value = duplicates.getProperty(key);
171                                key = key.replace(TIBCO_ALIAS_PREFIX, "");
172                                
173                                prefs.put(FILE_ALIAS_PREFIX + maxFileAliasPref.toString(), key + "=" + value);
174                                maxFileAliasPref++;
175                        }
176
177                        FileOutputStream fosPrefs = new FileOutputStream(designer5Prefs);
178                        prefs.store(fosPrefs, "");
179                        fis.close();
180
181                        aliases.putAll(duplicates);
182
183                        FileOutputStream fos = new FileOutputStream(aliasesFile);
184                        aliases.store(fos, "");
185                        fis.close();
186                }
187        }
