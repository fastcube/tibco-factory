017package com.github.fastcube.factory.tibco.bw.maven;
018
019import java.io.File;
020import java.io.IOException;
021
022import org.apache.maven.model.Model;
023import org.apache.maven.plugin.MojoExecutionException;
024import org.apache.maven.plugin.logging.Log;
025import org.apache.maven.plugins.annotations.LifecyclePhase;
026import org.apache.maven.plugins.annotations.Mojo;
027import org.apache.maven.project.MavenProject;
028import org.codehaus.plexus.util.xml.pull.XmlPullParserException;
029
030import com.github.fastcube.factory.tibco.bw.maven.source.POMManager;
031
032/**
033 * <p>
034 * This goal will initialize additional properties to use in next goals.
035 * It is the first goal called by the <b>bw-ear</b> lifecycle in the
036 * <b>initialize</b> phase.
037 * </p>
038 * 
039 * @author Radwane Mabchour
040 *
041 */
042@Mojo (name="initialize", defaultPhase=LifecyclePhase.INITIALIZE)
043public class InitializeMojo extends AbstractBWMojo {
044
045        protected static final String parentBasedirProperty = "parentProject.basedir";
046
047        public static boolean hasSameGAV(Model m1, Model m2) {
048                if (m1 == null || m2 == null) {
049                        return false;
050                } else {
051                        return m1.getGroupId().equals(m2.getGroupId()) &&
052                                   m1.getArtifactId().equals(m2.getArtifactId()) &&
053                                   (m1.getVersion().equals(m2.getVersion()) || m1.getVersion() == null || m2.getVersion() == null) && // tolerance if version is inherited
054                                   m1.getPackaging().equals(m2.getPackaging());                            
055                }
056        }
057
058        private static boolean propertyExistsInModel(Model model, String propertyKey) {
059                for (Object v : model.getProperties().values()) {
060                        String value = (String) v;
061                        
062                        if (value.contains("${" + propertyKey +"}")) {
063                                return true;
064                        }
065                }
066
067                return false;
068        }
069
070        private static boolean findParentPath(File path, Log logger) throws IOException, XmlPullParserException {
071                File pomPath = new File(path + File.separator + "pom.xml");
072                if (pomPath != null && pomPath.exists()) {
073                        logger.debug("pomPath: " + pomPath.getAbsolutePath());
074                        Model m = POMManager.getModelFromPOM(pomPath, logger);
075                        return propertyExistsInModel(m, parentBasedirProperty);
076                }
077                logger.debug("find false");
078                return false;
079        }
080
081        protected static File getParent(MavenProject project, Log logger) throws IOException, XmlPullParserException {
082                File result = null;
083                
084                MavenProject parent = project.getParent();
085                if (parent == null) {
086                        return result; // no parent: return null
087                }
088                
089                File parentPOM = parent.getFile();
090                File parentBasedir = null;
091                if (parentPOM != null &&
092                        parentPOM.getParentFile() != null &&
093                        parentPOM.getParentFile().exists() &&
094                        parentPOM.getParentFile().isDirectory()
095                        ) {
096                        parentBasedir = parentPOM.getParentFile();
097                }
098
099                if (parentPOM != null) {
100                        logger.debug("parentPOM: " + parentPOM.getAbsolutePath());
101                }
102
103                while (parentBasedir != null && parentBasedir.exists()) {
104                        logger.debug("parentBasedir: " + parentBasedir.getAbsolutePath());
105                        if (findParentPath(parentBasedir, logger)) {
106                                logger.debug("parentFound");
107                                result = parentBasedir;
108                                
109                                break;
110                        }
111                        logger.debug("parentNotFound");
112                        if (parent != null) {
113                                logger.debug(parent.getArtifactId());
114                                parentBasedir = parent.getParentFile(); // use <relativePath> to retrieve real parent file
115                                if (parentBasedir == null && parent.getParent() != null) {
116                                        parentBasedir = parent.getParent().getFile();
117                                }
118                                if (parentBasedir != null) {
119                                        logger.debug(parentBasedir.getAbsolutePath());
120                                }
121                                if (parentBasedir != null &&
122                                        parentBasedir.exists() &&
123                                        parentBasedir.isFile()) {
124                                        parentBasedir = parentBasedir.getParentFile();
125                                }
126                                parent = parent.getParent();
127                        }
128                }
129
130                return result;
131        }
132
133        private void setParentPath() throws IOException, XmlPullParserException {
134                File parentPath = getParent(getProject(), getLog());
135                if (parentPath != null) {
136                        getLog().debug(parentBasedirProperty + ": " + parentPath.getAbsolutePath());
137                        getProject().getProperties().setProperty(parentBasedirProperty, parentPath.getAbsolutePath());
138                }
139        }
140
141        @Override
142        public void execute() throws MojoExecutionException {
143                try {
144                        setParentPath();
145                } catch (Exception e) {
146                        throw new MojoExecutionException(e.getMessage(), e);
147                }
148        }
149
150}
