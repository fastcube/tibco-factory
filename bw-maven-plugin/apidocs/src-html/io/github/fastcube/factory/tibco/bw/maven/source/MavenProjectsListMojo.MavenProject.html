017package io.github.fastcube.factory.tibco.bw.maven.source;
018
019import java.io.File;
020import java.io.IOException;
021import java.lang.reflect.InvocationTargetException;
022import java.util.HashMap;
023
024import org.apache.maven.model.Model;
025import org.apache.maven.plugin.MojoExecutionException;
026import org.apache.maven.plugin.MojoFailureException;
027import org.apache.maven.plugin.logging.Log;
028import org.apache.maven.plugins.annotations.Mojo;
029import org.codehaus.plexus.util.xml.pull.XmlPullParserException;
030
031/**
032 * <p>
033 * This goal lists all the existing projects (with a POM.xml file) found in a
034 * specified folder and nested subfolders.<br />
035 * The mandatory 'pom.xml' file will be used as a search criteria.
036 * </p>
037 * 
038 * <p>
039 * Once the list of projects is ready, it is possible to use children goals such
040 * as <a href="./add-dependency-mojo.html">bw:add-dependency</a> with the same
041 * parameters.
042 * </p>
043 * 
044 * @author Radwane Mabchour
045 *
046 */
047@Mojo ( name = "list-maven-projects", requiresProject = false )
048public class MavenProjectsListMojo extends AbstractProjectsListMojo {
049
050        // these methods must be overriden in children classes
051        protected String getActionFailure() { return " action failed."; }
052        protected String getActionFailures() { return " actions failed."; }
053        protected String getActionSuccess() { return " action succeeded."; }
054        protected String getActionSuccesses() { return " actions succeeded."; }
055        
056        /**
057         * This concrete inner-class inheriting from AbstractProject will guess
058         * project information from the Maven model. This model is built from the
059         * POM of the project.
060         *
061         */
062        public static class MavenProject extends AbstractProject {
063                private Model model;
064                
065                @Override
066                public void initialize(File path, File root, HashMap<String, String> m, String mandatoryFilename, Log logger) throws IOException, XmlPullParserException {
067                        super.initialize(path, root, m, mandatoryFilename, logger);
068                        
069                        initModel();
070                }
071
072                private void initModel() throws IOException, XmlPullParserException {
073                        File pom = new File(getMandatoryFilePath());
074                        
075                        model = POMManager.getModelFromPOM(pom, logger);
076                }
077
078                @Override
079                public String getProjectName() {
080                        return model.getName();
081                }
082
083                @Override
084                public String getGroupId() {
085                        return model.getGroupId();
086                }
087                
088                @Override
089                public String getArtifactId() {
090                        return model.getArtifactId();
091                }
092
093                @Override
094                public String getVersion() {
095                        return model.getVersion() != null ? model.getVersion() : "[inherited]";
096                }
097        }
098
099        protected final static String MANDATORY_MAVEN_FILENAME = "pom.xml";
100 
101        @Override
102        protected void initProjects() throws IOException, InstantiationException, IllegalAccessException, IllegalArgumentException, SecurityException, InvocationTargetException, NoSuchMethodException, XmlPullParserException {
103                super.initProjects(MANDATORY_MAVEN_FILENAME, new HashMap<String, String>(), MavenProject.class);
104        }
105
106        @Override
107        protected void displayProject(AbstractProject p) {
108                getLog().info(p.getProjectName());
109                getLog().info("  Location of the project : " +
110                              p.getRelativePath()
111                             );
112                getLog().info("  Maven project is        : " +
113                                p.getGroupId() + ":" +
114                                p.getArtifactId() + ":" +
115                                p.getVersion()
116                                );
117        }
118        
119        @Override
120        public void execute() throws MojoExecutionException, MojoFailureException {
121                super.execute();
122                if (!doIt) {
123                        if (!this.getClass().getName().equals(MavenProjectsListMojo.class.getName())) {
124                                getLog().info(NOTHING_WAS_PERFORMED);
125                        }
126                } else if (projects.size() > 0) {
127                        int failed = projects.size() - successfullyPerformedAction;
128                        
129                        if (failed > 1) {
130                                getLog().info(failed + getActionFailures());
131                        } else {
132                                getLog().info(failed + getActionFailure());
133                        }
134                        if (successfullyPerformedAction > 1) {
135                                getLog().info(successfullyPerformedAction + getActionSuccesses());
136                        } else {
137                                getLog().info(successfullyPerformedAction + getActionSuccess());
138                        }
139                }
140        }
141
142}
