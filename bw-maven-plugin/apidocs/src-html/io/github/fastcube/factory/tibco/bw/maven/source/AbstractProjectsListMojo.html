038/**
039 * <p>
040 * This is the base class to define a list of projects. The list will be
041 * populating according to several criteria such as {@code workspaceRoot}
042 * (directory where to look for the projects) or the type of project defined by
043 * concrete children classes of inner-class {@link AbstractProject}.
044 * </p>
045 * 
046 * @author Radwane Mabchour
047 *

051        /**
052         * <p>
053         * A project can be either:
054         * <ul>
055         * <li>an existing Maven project defined by a POM</li>
056         * <li>an existing BW project that is to be mavenized (Maven properties such
057         * as groupId, artifactId & version will be guessed from provided values)
058         * </li>
059         * </ul>
060         * Concrete inner-classes will define the properties according to these
061         * profiles.
062         * </p>
063         */
064        public static abstract class AbstractProject {
065                protected Log logger; // Maven logger (because the inner-class must be static)
066                private String mandatoryFilename; // the mandatory filename will trigger the add of a project to the projects list
067                private File path; // original path of the file that triggered the creation of an AbstractProject
068                protected File root; // root path where the projects are searched
069
070                /**
071                 * The concrete inner-classes must be instantiated with the default
072                 * constructor (with no arguments).
073                 * The instantiator is responsible for calling the initialize method
074                 * right after the instantiation of concrete inner-class.
075                 * @throws XmlPullParserException 
076                 * @throws IOException 
077                 */
078                public void initialize(File path, File root, HashMap<String, String> m, String mandatoryFilename, Log logger) throws IOException, XmlPullParserException {
079                        this.logger = logger;
080                        this.mandatoryFilename = mandatoryFilename;
081                        this.path = path;
082                        this.root = root;                       
083                }
084
085                public String getAbsolutePath() {
086                        return root + File.separator + getRelativePath();
087//                      return root + File.separator + getOriginalRelativePath();
088                }
089
090                public String getOriginalAbsolutePath() {
091//                      return root + File.separator + getRelativePath();
092                        return root + File.separator + getOriginalRelativePath();
093                }
094
095                public String getMandatoryFilePath() {
096                        return getOriginalAbsolutePath() + File.separator + mandatoryFilename;
097                }
098                
099                public String getMandatoryFilePath(String customMandatoryFilename) {
100                        return getOriginalAbsolutePath() + File.separator + customMandatoryFilename;
101                }
102                
103                private String _getRelativePath() {
104                        String s = path.getAbsolutePath();
105                        s = s.replace(root.getAbsolutePath()+File.separator, "");
106
107                        s = s.replace(mandatoryFilename, "");
108                        if (s.endsWith(File.separator)) {
109                                s = s.substring(0, s.length()-1);
110                        }
111                        
112                        return s;
113                }
114                public String getOriginalRelativePath() {
115                        return _getRelativePath();
116                }
117                
118                public String getRelativePath() {
119                        return _getRelativePath();
120                }
121
122                // default is not to ignored (concrete inner-classes can override)
123                public boolean isIgnored() {
124                        return false;
125                }
126                
127                // information of the project from Maven point-of-view
128                public abstract String getProjectName();
129                public abstract String getGroupId();
130                public abstract String getArtifactId();
131                public abstract String getVersion();
132
133                public Boolean getPutPOMInParentDirectory() {
134                        return false;
135                }
136        }

182            super(msg);
183        }

186    /**
187     * <p>
188     * Get the relative path from one file to another, specifying the directory
189     * separator. 
190     * If one of the provided resources does not exist, it is assumed to be a
191     * file unless it ends with '/' or '\'.
192     * </p>
193     * 
194     * @param targetPath targetPath is calculated to this file
195     * @param basePath basePath is calculated from this file
196     * @param pathSeparator directory separator. The platform default is not assumed so that we can test Unix behaviour when running on Windows (for example)
197     * @return
198     */
199    public static String getRelativePath(String targetPath, String basePath, String pathSeparator) {
200        // Normalize the paths
201        String normalizedTargetPath = FilenameUtils.normalizeNoEndSeparator(targetPath);
202        String normalizedBasePath = FilenameUtils.normalizeNoEndSeparator(basePath);
203
204        // Undo the changes to the separators made by normalization
205        if (pathSeparator.equals("/")) {
206            normalizedTargetPath = FilenameUtils.separatorsToUnix(normalizedTargetPath);
207            normalizedBasePath = FilenameUtils.separatorsToUnix(normalizedBasePath);
208
209        } else if (pathSeparator.equals("\\")) {
210            normalizedTargetPath = FilenameUtils.separatorsToWindows(normalizedTargetPath);
211            normalizedBasePath = FilenameUtils.separatorsToWindows(normalizedBasePath);
212
213        } else {
214            throw new IllegalArgumentException("Unrecognised dir separator '" + pathSeparator + "'");
215        }
216
217        String[] base = normalizedBasePath.split(Pattern.quote(pathSeparator));
218        String[] target = normalizedTargetPath.split(Pattern.quote(pathSeparator));
219
220        // First get all the common elements. Store them as a string,
221        // and also count how many of them there are.
222        StringBuffer common = new StringBuffer();
223
224        int commonIndex = 0;
225        while (commonIndex < target.length && commonIndex < base.length
226                && target[commonIndex].equals(base[commonIndex])) {
227            common.append(target[commonIndex] + pathSeparator);
228            commonIndex++;
229        }
230
231        if (commonIndex == 0) {
232            // No single common path element. This most
233            // likely indicates differing drive letters, like C: and D:.
234            // These paths cannot be relativized.
235            throw new PathResolutionException("No common path element found for '" + normalizedTargetPath + "' and '" + normalizedBasePath + "'");
236        }   
237
238        // The number of directories we have to backtrack depends on whether the base is a file or a dir
239        // For example, the relative path from
240        //
241        // /foo/bar/baz/gg/ff to /foo/bar/baz
242        // 
243        // ".." if ff is a file
244        // "../.." if ff is a directory
245        //
246        // The following is a heuristic to figure out if the base refers to a file or dir. It's not perfect, because
247        // the resource referred to by this path may not actually exist, but it's the best I can do
248        boolean baseIsFile = true;
249
250        File baseResource = new File(normalizedBasePath);
251
252        if (baseResource.exists()) {
253            baseIsFile = baseResource.isFile();
254
255        } else if (basePath.endsWith(pathSeparator)) {
256            baseIsFile = false;
257        }
258
259        StringBuffer relative = new StringBuffer();
260
261        if (base.length != commonIndex) {
262            int numDirsUp = baseIsFile ? base.length - commonIndex - 1 : base.length - commonIndex;
263
264            for (int i = 0; i < numDirsUp; i++) {
265                relative.append(".." + pathSeparator);
266            }
267        }
268        relative.append(normalizedTargetPath.substring(common.length()));
269        return relative.toString();
270    }

272    private static String toCommaSeparatedString(List<String> strings) {
273        StringBuilder sb = new StringBuilder();
274        for (String string : strings) {
275            if (sb.length() > 0) {
276                sb.append(", ");
277            }
278            sb.append(string);
279        }
280        return sb.toString();
281    }

283        public static List<File> toFileList(FileSet fileSet) throws IOException {
284        File directory = new File(fileSet.getDirectory());
285        String includes = toCommaSeparatedString(fileSet.getIncludes());
286        String excludes = toCommaSeparatedString(fileSet.getExcludes());
287        return FileUtils.getFiles(directory, includes, excludes);
288    }

296        protected abstract void initProjects() throws IOException, InstantiationException, IllegalAccessException, IllegalArgumentException, SecurityException, InvocationTargetException, NoSuchMethodException, XmlPullParserException;
297        
298        protected final void initProjects(String mandatoryFilename, HashMap<String, String> m, Class<? extends AbstractProject> classAbstractProject) throws IOException, InstantiationException, IllegalAccessException, IllegalArgumentException, InvocationTargetException, SecurityException, NoSuchMethodException, XmlPullParserException {
299                projects = new ArrayList<AbstractProject>();
300                
301                restriction= new FileSet();
302                restriction.setDirectory(workspaceRoot.getAbsolutePath());
303                
304                if (recursive) {
305                        for (String p : patterns) {
306                                restriction.addInclude(p + mandatoryFilename);
307                        }
308                } else {
309                        restriction.addInclude(mandatoryFilename); // we are only looking in the "workspaceRoot" directory
310                }
311                restriction.addExclude("**/target/**/" + mandatoryFilename);
312                restriction.addExclude("**/bin/" + mandatoryFilename);
313                
314                List<File> files = toFileList(restriction);
315                
316                for (File f : files) {
317                        Constructor<? extends AbstractProject> ctor = classAbstractProject.getConstructor();
318                        AbstractProject ap = ctor.newInstance();
319                        
320                        ap.initialize(f, workspaceRoot, m, mandatoryFilename, getLog());
321                        if (!ap.isIgnored()) {
322                                projects.add(ap);
323                        } else {
324                                getLog().debug("Ignoring project '" + ap.getProjectName() + "'");
325                        }
326                }
327        }
328        
329        protected void browseList(){
330                for (AbstractProject p : projects) {
331                        // display
332                        if (displayList) {
333                                getLog().info("---");
334                                displayProject(p);
335                                getLog().info("");
336                        }
337                        
338                        // perform action (if exists)
339                        if (doIt && performAction(p)) {
340                                successfullyPerformedAction++;
341                        }
342                }
343                
344                getLog().info("");
345                if (projects.size()>1) {
346                        getLog().info(projects.size() + PROJECTS_FOUND);
347                } else {
348                        getLog().info(projects.size() + PROJECT_FOUND);
349                }
350                
351        }
352
353        public void execute() throws MojoExecutionException, MojoFailureException {
354                if (patterns == null || patterns.isEmpty()) {
355                        patterns = new ArrayList<String>();
356                        patterns.add("**/");
357                }
358                
359                if (recursive) {
360                        getLog().info("Scanning recursively '" + workspaceRoot + "' with patterns '" + patterns.toString() +"'");
361                } else {
362                        getLog().info("Scanning '" + workspaceRoot + "'");
363                }
364                
365                successfullyPerformedAction = 0;
366                try {
367                        initProjects();
368                } catch (Exception e) {
369                        throw new MojoExecutionException(e.getLocalizedMessage(), e);
370                }
371                
372                browseList();
373        }
