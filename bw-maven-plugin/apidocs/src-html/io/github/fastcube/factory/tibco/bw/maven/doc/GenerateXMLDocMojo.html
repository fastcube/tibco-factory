022import javax.xml.bind.JAXBException;
023import javax.xml.bind.Marshaller;
024
025import org.apache.maven.plugin.MojoExecutionException;

027import org.apache.maven.plugins.annotations.LifecyclePhase;
028import org.apache.maven.plugins.annotations.Mojo;
029import org.w3c.dom.Element;
030import org.w3c.dom.Node;
031import org.w3c.dom.NodeList;

033import com.tibco.xmlns.bw.process._2003.Activity;
034import com.tibco.xmlns.bw.process._2003.Label;
035import com.tibco.xmlns.bw.process._2003.ProcessDefinition;
036
037import io.github.fastcube.factory.tibco.bw.maven.doc.jaxb.Bwdoc;
038import io.github.fastcube.factory.tibco.bw.maven.doc.jaxb.Bwdoc.CalledProcesses;
039import io.github.fastcube.factory.tibco.bw.maven.doc.jaxb.Bwdoc.Labels;
040import io.github.fastcube.factory.tibco.bw.maven.doc.jaxb.ObjectFactory;
041import io.github.fastcube.factory.tibco.bw.maven.doc.jaxb.Process;

045/**
046 * <p>
047 * This goal generates an XML documentation file from TIBCO BusinessWorks source
048 * files.
049 * </p>
050 * 
051 * @author Radwane Mabchour
052 * 
053 */
054@Mojo( name="generate-doc-xml",
055defaultPhase=LifecyclePhase.PROCESS_SOURCES )
056@Execute( goal="generate-doc-xml", phase=LifecyclePhase.PROCESS_SOURCES)
057public class GenerateXMLDocMojo extends AbstractDocMojo {
058
059        private File xmlDirectory;
060        
061        private ProcessDefinition getProcessModel(File f) throws MojoExecutionException {
062                ProcessDefinition result = null;
063                try {
064                        ProcessModel pm = new ProcessModel(f);
065                        result = pm.getProcess();
066                } catch (JAXBException e) {
067                        throw new MojoExecutionException(e.getMessage(), e);
068                }
069                return result;
070        }
071        
072        @Override
073        public void processFile(File f) throws MojoExecutionException {
074                ProcessDefinition pd = getProcessModel(f);
075                if (pd == null) return;
076
077                String name = AbstractProjectsListMojo.getRelativePath(f.getAbsolutePath(),
078                                                                              getDirectoryToProcess().getAbsolutePath(),
079                                                                              File.separator);
080
081                //File out = new File(xmlDirectory + File.separator + f.getName() + ".xml");
082                File out = new File(xmlDirectory + File.separator + name + ".xml");
083                out.getParentFile().mkdirs();
084                
085                getLog().debug("Processing '" + f.getAbsolutePath() + "' to '" + out.getAbsolutePath());
086
087                // description of the process
088                Bwdoc doc = new Bwdoc();
089                doc.setDescription(pd.getDescription());
090
091                if (doc.getLabels() == null) {
092                        doc.setLabels(new Labels());
093                }
094
095                for (Label l : pd.getLabel()) {
096                        io.github.fastcube.factory.tibco.bw.maven.doc.jaxb.Label label = new io.github.fastcube.factory.tibco.bw.maven.doc.jaxb.Label();
097                        label.setName(l.getName());
098                        label.setDescription(l.getDescription());
099                        doc.getLabels().getLabel().add(label);
100                }
101
102                if (doc.getCalledProcesses() == null) {
103                        doc.setCalledProcesses(new CalledProcesses());
104                }
105
106                // retrieve "Call Process" activities
107                for (Activity a : pd.getActivity()) {
108                        if (a.getType().equals("com.tibco.pe.core.CallProcessActivity")) {
109                                String pathValue = "";
110                                Element configElement = (Element) a.getConfig();
111                                NodeList nodes = configElement.getElementsByTagName("processName");
112                                if (nodes.getLength() > 0) {
113                                        Node node = nodes.item(0).getFirstChild();
114                                        if (node != null) {
115                                                pathValue = node.getNodeValue();
116                                        }
117                                }
118                                Process p = new Process();
119                                p.setName(a.getName());
120                                p.setPath(pathValue);
121                                doc.getCalledProcesses().getCalledProcess().add(p);
122                        }
123                } 
124                //
125                
126                try {
127                        save(doc, out);
128                } catch (JAXBException e) {
129                        throw new MojoExecutionException(e.getMessage(), e);
130                }
131        }
132
133        @Override
134        public void execute() throws MojoExecutionException {
135                initXMLDirectory();
136                
137                super.execute();
138        }
139        
140        private void initXMLDirectory() {
141                xmlDirectory = new File(docDirectory + File.separator + "xml");
142
143                getLog().debug(xmlDirectory.getAbsolutePath());
144                if (!xmlDirectory.exists()) {
145                        xmlDirectory.mkdirs();
146                }               
147        }
148
149        private void save(Bwdoc doc, File f) throws JAXBException {
150                JAXBContext jaxbContext = JAXBContext.newInstance(ObjectFactory.class);
151                Marshaller m = jaxbContext.createMarshaller();
152                m.setProperty(Marshaller.JAXB_FORMATTED_OUTPUT, true);          
153                m.marshal(doc, f);
154        }
155        
156}
