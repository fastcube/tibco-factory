041/**
042 * <p>
043 * This goal will merge the reference and common properties into the working
044 * properties files (usually the ones in "target/package").
045 * </p>
046 * 
047 * @author Radwane Mabchour
048 *
049 */
050@Mojo( name="merge-properties",
051defaultPhase=LifecyclePhase.PREPARE_PACKAGE )
052public class MergePropertiesMojo extends AbstractPackagingMojo {
053
054        /**
055         * By default, the empty binding ("bw[EXAMPLE.par]/bindings/binding[]) is
056         * removed if other named bindings exists
057         * ("bw[EXAMPLE.par]/bindings/binding[named]").
058         * It is possible to keep this empty binding by setting
059     * <i>alwaysKeepEmptyBindings</i> to <i>true</i>. Default is <i>false</i>.
060         */
061        @Parameter (property = "alwaysKeepEmptyBindings", defaultValue = "false")
062        boolean alwaysKeepEmptyBindings;
063        
064        /**
065         * It is possible to ignore any merging of properties by setting
066         * <i>ignorePropertiesMerge</i> to <i>true</i>. Default is <i>false</i>
067         * (do merge).
068         */
069        @Parameter (property = "ignorePropertiesMerge", defaultValue = "false")
070        boolean ignorePropertiesMerge;
071
072        /**
073         * By default the properties from common properties are merged first and the
074         * properties from reference specific to the project last. By setting 
075         * <i>mergeCommonLast</i> to true, the common properties are mergerd last.
076         */
077        @Parameter (property = "mergeCommonLast", defaultValue = "false")
078        boolean mergeCommonLast; 
079        
080        private enum MergedFiles {
081                NONE, GV, SERVICES, BOTH
082        }
083        
084        protected final static String MERGING_PROPERTIES = "Merging properties...";
085        protected final static String NOTHING_TO_MERGE = "Nothing to merge.";
086        protected final static String MERGE_FAILURE = "Failed to merge properties";
087        protected final static String PROPERTIES_LOAD_FAILURE = "Failed to load properties";
088
089        protected final static String PROPERTIES_SAVE_GVS_SUCCESS = "Successfully saved merged properties to file";
090        protected final static String PROPERTIES_SAVE_GVS_FAILURE = "Failed to save merged properties to XML file";
091        protected final static String PROPERTIES_SAVE_SERVICES_SUCCESS = "Successfully saved merged properties to file";
092        protected final static String PROPERTIES_SAVE_SERVICES_FAILURE = "Failed to save merged properties to XML file";
093
094        protected final static String USING_GVS_COMMON_FILE = "Using Global Variables common properties file";
095        protected final static String USING_GVS_REFERENCE_FILE = "Using Global Variables reference properties file";
096        protected final static String USING_SERVICES_COMMON_FILE = "Using Services common properties file";
097        protected final static String USING_SERVICES_REFERENCE_FILE = "Using Services reference properties file";

099        private List<ImmutablePair<String, String>> pairParInstance = new ArrayList<ImmutablePair<String, String>>(); // keep trace of dynamically created bindings (to generate them only once)
100
101        private final static String regexNotEmptyBinding = "^bw\\[(.*)\\]/bindings/binding\\[(.+)\\]/(.*)$";
102        private final static String regexEmptyBinding = "^bw\\[(.*)\\]/bindings/binding(\\[\\])/(.*)$";
103
104        @Override
105        protected String getArtifactFileExtension() {
106                return PROPERTIES_EXTENSION;
107        }

109        public MergedFiles mergeFiles(Properties propertiesGlobalVariables,
110                                                                  Properties propertiesServices, 
111                                                                  File deploymentGlobalVariables,
112                                                                  File deploymentServices,
113                                                                  String messageGlobalVariables,
114                                                                  String messageServices) throws MojoExecutionException, IOException {
115
116                MergedFiles result = MergedFiles.NONE;
117                
118                // Merge with files if they exist
119                if (deploymentGlobalVariables != null &&
120                        deploymentGlobalVariables.exists() &&
121                        !deploymentGlobalVariables.getCanonicalPath().equals(this.deploymentGlobalVariables.getCanonicalPath())) { 
122                        // FIXME: check the checksum of files instead of canonical path ?
123                        getLog().info(messageGlobalVariables + " : " + deploymentGlobalVariables.getAbsolutePath());
124                        
125                        Properties propertiesGlobalVariablesReference;
126                        try {
127                                propertiesGlobalVariablesReference = loadPropertiesFile(deploymentGlobalVariables); 
128                        } catch (Exception e) {
129                                throw new MojoExecutionException(PROPERTIES_LOAD_FAILURE, e);
130                        }
131                        
132                        Enumeration<Object> e = propertiesGlobalVariablesReference.keys();
133                        while (e.hasMoreElements()) {
134                                String key = (String) e.nextElement();
135                                String value = propertiesGlobalVariablesReference.getProperty(key);
136                                
137                                propertiesGlobalVariables.setProperty(key, value); // add or override value
138                        }
139                        
140                        result = MergedFiles.GV;
141                }
142                
143                Pattern pNotEmptyBinding = Pattern.compile(regexNotEmptyBinding);
144                
145                if (deploymentServices != null &&
146                        deploymentServices.exists() &&
147                        !deploymentServices.getCanonicalPath().equals(this.deploymentServices.getCanonicalPath())) {
148                        // FIXME: check the checksum of files instead of canonical path ?
149                        
150                        getLog().info(messageServices + " : " + deploymentServices.getAbsolutePath());
151                        Properties propertiesServicesReference;
152                        try {
153                                propertiesServicesReference = loadPropertiesFile(deploymentServices);
154                        } catch (Exception e) {
155                                throw new MojoExecutionException(PROPERTIES_LOAD_FAILURE, e);
156                        }
157                        
158                        String par, binding;
159                        Enumeration<Object> e = propertiesServicesReference.keys();
160                        while (e.hasMoreElements()) {
161                                String key = (String) e.nextElement();
162                                
163                                Matcher mNotEmptyBinding = pNotEmptyBinding.matcher(key);
164                                
165                                if (mNotEmptyBinding.matches() && !isAWildCard(key)) {
166                                        par = mNotEmptyBinding.group(1);
167                                        binding = mNotEmptyBinding.group(2);
168                                        ImmutablePair<String, String> pair = new ImmutablePair<String, String>(par, binding);
169                                        if (!pairParInstance.contains(pair)) {
170                                                propertiesServices = duplicateEmptyBinding(propertiesServices, par, binding);
171                                                pairParInstance.add(pair);
172                                        }
173                                }
174                                
175                                String value = propertiesServicesReference.getProperty(key);
176                                
177                                propertiesServices.setProperty(key, value); // add or override value
178                        }
179                        
180                        propertiesServices = expandWildCards(propertiesServices);
181                        
182                        if (result == MergedFiles.GV) {
183                                result = MergedFiles.BOTH;
184                        } else {
185                                result = MergedFiles.SERVICES;
186                        }
187                }
188                
189                return result;
190        }
191
192        private Properties duplicateEmptyBinding(Properties properties, String par, String binding) {
193                String key, value;
194                Pattern pEmptyBinding = Pattern.compile(regexEmptyBinding);
195
196                Enumeration<Object> e = properties.keys();
197                while (e.hasMoreElements()) {
198                        key = (String) e.nextElement();
199                        value = properties.getProperty(key);
200                        Matcher mEmptyBinding = pEmptyBinding.matcher(key);
201                        if (mEmptyBinding.matches() && mEmptyBinding.group(1).equals(par)) {
202                                key = "bw[" + mEmptyBinding.group(1) + "]/bindings/binding[" + binding + "]/" + mEmptyBinding.group(3);
203                        }
204                        properties.setProperty(key, value);
205                }
206                return properties;
207        }

209        private Properties sortProperties(Properties properties) throws ConfigurationException, IOException {
210                Properties sp = new SortedProperties();
211                sp.putAll(properties);
212                
213                ByteArrayOutputStream baos = new ByteArrayOutputStream();
214                sp.store(baos, null);
215                
216                properties.clear();
217                properties.load(new ByteArrayInputStream(baos.toByteArray()));
218
219                return properties;
220        }

222    protected boolean isAWildCard(String key) {
223                String regexVariable = "bw\\[[^\\**]*\\]/variables\\[.*\\]/variable\\[.*\\]"; // is a wildcard only if it has a '*' in bw[] for variables (because a variable can have '*' in their name)
224
225                return key.contains("*") && !Pattern.matches(regexVariable, key);
226    }

228    /**
229     * <p>
230     * This expands wild cards properties.<br /><br />
231     * Both wildcard expressions and expressions to expand are present in the
232     * same properties object.<br /><br />
233     * <i>Example</i>
234     *  <ul>
235     *   <li><b>property with wildcard</b>: /root/element[*]/key=new_value</li>
236     *   <li><b>property matching</b>: /root/element[my_name]/key=old_value</li>
237     *  </ul>
238     *  will expand to:<br />
239     *  <ul>
240     *   <li><b>property after expansion</b>:
241     * /root/element[my_name]/key=new_value</li>
242     *  </ul>
243     * </p>
244     * 
245     * @param properties, the properties object with wildcard expressions and
246     * expressions to expand
247     * @return properties with expanded expressions, but without wildcard
248     * expressions
249     */
250    protected Properties expandWildCards(Properties properties) {
251        Properties propertiesWithWildCards = new Properties() { // sorted properties
252                        private static final long serialVersionUID = 7793482336210629858L;
253
254                        @Override
255                    public synchronized Enumeration<Object> keys() {
256                        return Collections.enumeration(new TreeSet<Object>(super.keySet()));
257                    }
258                };
259        String key;
260        
261        // retrieve the keys with WildCards
262                Enumeration<Object> e = properties.keys();
263                while (e.hasMoreElements()) {
264                key = (String) e.nextElement();
265                if (isAWildCard(key)) {
266                        propertiesWithWildCards.setProperty(key, properties.getProperty(key));
267                        properties.remove(key);
268                }
269        }
270        
271        // try to replace the values of other keys matching the keys with WildCards
272                Enumeration<Object> w = propertiesWithWildCards.keys();
273                while (w.hasMoreElements()) {
274                String keyWithWildCards = (String) w.nextElement();
275                String regex = wildcardToRegex(keyWithWildCards);
276
277                        String ignoreWildcardInVariablesPattern = "(.*)variables\\\\\\[(.*)\\\\\\]\\/variable\\\\\\[(.*)\\\\\\](.*)";
278                        Pattern p = Pattern.compile(ignoreWildcardInVariablesPattern);
279                        Matcher m = p.matcher(regex);
280                        if (m.matches()) {
281                                String variables = m.group(2);
282                                String variable = m.group(3);
283                                variables = variables.replace(".*", "\\*");
284                                variable = variable.replace(".*", "\\*");
285                                regex = m.group(1) + "variables\\[" + variables + "\\]/variable\\[" + variable + "\\]" + m.group(4);
286                        }
287
288                Boolean found = false;
289
290                e = properties.keys();
291                while (e.hasMoreElements()) {
292                        key = (String) e.nextElement();
293                        
294                        if (Pattern.matches(regex, key)) {
295                                found = true;
296                                String value = (String) propertiesWithWildCards.getProperty(keyWithWildCards);
297                                properties.setProperty(key, value);
298                        }
299                }
300
301                // not found, we put back the expression with wild cards in the original list (false positive)
302                // this way the wildcard can still be used in a next pass and will be removed at the end by AbstractPackagingMojo.removeWildCards 
303                if (!found) {
304                        properties.setProperty(keyWithWildCards, propertiesWithWildCards.getProperty(keyWithWildCards));
305                }
306        }
307        
308        return properties;
309    }
310
311    protected Properties removeWildCards(Properties properties) {
312        String key;
313        
314                Enumeration<Object> e = properties.keys();
315        while (e.hasMoreElements()) {
316                key = (String) e.nextElement();
317                
318                if (isAWildCard(key)) {
319                        properties.remove(key);
320                }
321        }
322        
323        return properties;
324    }

326        public void execute() throws MojoExecutionException {
327                if (super.skip()) {
328                        return;
329                }
330
331                if (ignorePropertiesMerge) return; // ignore all properties merging
332
333                getLog().info(MERGING_PROPERTIES);
334                try {
335                        Properties earGlobalVariables = loadPropertiesFile(deploymentGlobalVariables);
336                        Properties earServices = loadPropertiesFile(deploymentServices);
337
338                        MergedFiles mergedCommonFiles = MergedFiles.NONE, mergedReferenceFiles = MergedFiles.NONE;
339
340                        if (mergeCommonLast) {
341                                if (!ignoreReferenceFiles) {
342                                        mergedReferenceFiles = mergeFiles(earGlobalVariables,
343                                                                                                          earServices,
344                                                                                                          deploymentGlobalVariablesReference,
345                                                                                                          deploymentServicesReference,
346                                                                                                          USING_GVS_REFERENCE_FILE,
347                                                                                                          USING_SERVICES_REFERENCE_FILE);
348                                }
349                                if (!ignoreCommonFiles) {
350                                        mergedCommonFiles = mergeFiles(earGlobalVariables,
351                                                                                                   earServices,
352                                                                                                   deploymentGlobalVariablesCommon,
353                                                                                                   deploymentServicesCommon,
354                                                                                                   USING_GVS_COMMON_FILE,
355                                                                                                   USING_SERVICES_COMMON_FILE);
356                                }
357                        } else {
358                                if (!ignoreCommonFiles) {
359                                        mergedCommonFiles = mergeFiles(earGlobalVariables,
360                                                                                                   earServices,
361                                                                                                   deploymentGlobalVariablesCommon,
362                                                                                                   deploymentServicesCommon,
363                                                                                                   USING_GVS_COMMON_FILE,
364                                                                                                   USING_SERVICES_COMMON_FILE);
365                                }
366                                if (!ignoreReferenceFiles) {
367                                        mergedReferenceFiles = mergeFiles(earGlobalVariables,
368                                                                                                          earServices,
369                                                                                                          deploymentGlobalVariablesReference,
370                                                                                                          deploymentServicesReference,
371                                                                                                          USING_GVS_REFERENCE_FILE,
372                                                                                                          USING_SERVICES_REFERENCE_FILE);
373                                }
374                        }
375
376                    earGlobalVariables = sortProperties(earGlobalVariables);
377                    earServices = sortProperties(earServices);
378
379                    earServices = removeWildCards(earServices);
380                    if (!alwaysKeepEmptyBindings) {
381                        earServices = removeEmptyBindings(earServices);
382                    }
383
384
385                        if (mergedCommonFiles == MergedFiles.GV ||
386                            mergedCommonFiles == MergedFiles.BOTH ||
387                            mergedReferenceFiles == MergedFiles.GV ||
388                            mergedReferenceFiles == MergedFiles.BOTH ||
389                            mergedCommonFiles == MergedFiles.SERVICES ||
390                            mergedCommonFiles == MergedFiles.BOTH ||
391                            mergedReferenceFiles == MergedFiles.SERVICES ||
392                            mergedReferenceFiles == MergedFiles.BOTH) {
393
394                                getLog().info("");
395                        } else {
396                                getLog().debug("GV ref: " + deploymentGlobalVariablesReference.getAbsolutePath());
397                                getLog().debug("SVC ref: " + deploymentServicesReference.getAbsolutePath());
398                                getLog().debug("GV common: " + deploymentGlobalVariablesCommon.getAbsolutePath());
399                                getLog().debug("SVC common: " + deploymentServicesCommon.getAbsolutePath());
400                                getLog().info(NOTHING_TO_MERGE);
401                        }
402
403                        // Export Properties in a properties file
404                        //  Global Variables
405                        if (mergedCommonFiles == MergedFiles.GV ||
406                            mergedCommonFiles == MergedFiles.BOTH ||
407                            mergedReferenceFiles == MergedFiles.GV ||
408                            mergedReferenceFiles == MergedFiles.BOTH) {
409                                savePropertiesToFile(deploymentGlobalVariables,
410                                                                         earGlobalVariables,
411                                                                         "Global Variables",
412                                                                         PROPERTIES_SAVE_GVS_SUCCESS,
413                                                                         PROPERTIES_SAVE_GVS_FAILURE);
414                        }
415                        //  Services (=~ Process Archives)
416                        if (mergedCommonFiles == MergedFiles.SERVICES ||
417                            mergedCommonFiles == MergedFiles.BOTH ||
418                            mergedReferenceFiles == MergedFiles.SERVICES ||
419                            mergedReferenceFiles == MergedFiles.BOTH) {
420                                savePropertiesToFile(deploymentServices,
421                                                                         earServices,
422                                                                         "Services (Bindings, Processes)",
423                                                                         PROPERTIES_SAVE_SERVICES_SUCCESS,
424                                                                         PROPERTIES_SAVE_SERVICES_FAILURE);
425                        }
426                } catch (Exception e) {
427                        throw new MojoExecutionException(MERGE_FAILURE, e);
428                }
429        }

431        private Properties removeEmptyBindings(Properties properties) {
432                ArrayList<String> pars = new ArrayList<String>();
433                
434                Pattern pNotEmptyBinding = Pattern.compile(regexNotEmptyBinding);
435                Pattern pEmptyBinding = Pattern.compile(regexEmptyBinding);
436                String parName;
437                
438                Enumeration<Object> e = properties.keys();
439                // first check if there is at least one non empty binding (non default)
440                while (e.hasMoreElements()) {
441                        String key = (String) e.nextElement();
442                        Matcher mNotEmptyBinding = pNotEmptyBinding.matcher(key);
443                        if (mNotEmptyBinding.matches()) {
444                                parName = mNotEmptyBinding.group(1);
445                                if (pars.contains(parName)) continue;
446                                pars.add(parName);
447                        }
448                }
449                // then delete
450                e = properties.keys();
451                while (e.hasMoreElements()) {
452                        String key = (String) e.nextElement();
453                        Matcher mEmptyBinding = pEmptyBinding.matcher(key);
454                        if (mEmptyBinding.matches()) {
455                                parName = mEmptyBinding.group(1);
456                                if (pars.contains(parName)) {
457                                        properties.remove(key);
458                                }
459                        }
460                }
461                
462                return properties;
463        }
