017package io.github.fastcube.factory.tibco.bw.maven.compile;
018
019import java.io.File;
020import java.io.IOException;
021import java.util.ArrayList;
022
023import org.apache.maven.plugin.MojoExecutionException;
024import org.apache.maven.plugins.annotations.LifecyclePhase;
025import org.apache.maven.plugins.annotations.Mojo;
026import org.apache.maven.plugins.annotations.Parameter;
027
028import io.github.fastcube.factory.tibco.bw.maven.AbstractBWArtifactMojo;
029import io.github.fastcube.factory.tibco.bw.maven.AbstractBWMojo;
030import io.github.fastcube.factory.tibco.bw.maven.exception.BinaryMissingException;
031
032/**
033 * <p>
034 * This goal creates a TIBCO BusinessWorks Projib based on the configuration
035 * given in the project's POM.
036 * </p>
037 * <p>
038 * The most important parameters are:
039 * </p>
040 * <ul>
041 *      <li><b>bw.project.library.builder</b> : the location of the Library
042 * Builder file inside the TIBCO BusinessWorks project</li>
043 *      <li><b>bw.project.location</b> : the TIBCO BusinessWorks sources
044 * location relatively to the POM</li>
045 * </ul>
046 * <p>
047 * Other parameters have default values set with the global values of
048 * "<a href="./documentation/configuration.html#Maven_settingsxml_file">settings.xml</a>"
049 * file.
050 * </p>
051 * 
052 * @author Radwane Mabchour
053 *
054 */
055@Mojo( name="compile-projlib",
056defaultPhase=LifecyclePhase.COMPILE )
057public class CompileProjlibMojo extends AbstractBWArtifactMojo {
058
059        protected final static String TIBCO_DESIGNER_BINARY_NOTFOUND = "The TIBCO Designer binary can't be found.";
060        protected final static String BUILD_PROJLIB_FAILED = "The build of the Projlib file has failed.";
061        protected final static String BUILDING_PROJLIB = "Building the Projlib...";
062
063        /**
064         * Path to the LibBuilder relatively to the BusinessWorks project path.
065         */
066        @Parameter ( property="bw.project.library.builder", required = true)
067        private String libBuilder;
068
069        /**
070         * Path to the TIBCO Designer "buildlibrary" binary.
071         * 
072         */
073        @Parameter ( property="buildlibrary.path", required = true)
074        private File tibcoBuildLibraryPath;
075            
076        private void checkDesignerBuildLibrary() throws MojoExecutionException {
077                if (tibcoBuildLibraryPath == null ||
078                   !tibcoBuildLibraryPath.exists() ||
079                   !tibcoBuildLibraryPath.isFile()) {
080                        designerBuildLibraryPathNotFound();
081                }               
082        }
083
084        private void designerBuildLibraryPathNotFound() throws MojoExecutionException {
085                throw new BinaryMissingException(TIBCO_DESIGNER_BINARY_NOTFOUND);               
086        }
087
088        protected String getArtifactFileExtension() {
089                return PROJLIB_EXTENSION;
090        }
091        
092        /**
093         *  This calls the "buildlibrary" binary found in TIBCO Designer to build
094         * a Projlib for the {@link AbstractBWMojo#project}, defined by the {@link CompileProjlibMojo#libBuilder}.
095         * 
096         * @param outputFile, the path where the Projlib output will be created
097         * @throws MojoExecutionException
098         * @throws IOException
099         */
100        private void buildProjlib(File outputFile) throws MojoExecutionException, IOException {
101                checkDesignerBuildLibrary();
102                assert(outputFile != null);
103
104                ArrayList<String> arguments = new ArrayList<String>();          
105                arguments.add("-lib"); // path of the libbuilder relative to the BW project path
106                arguments.add(libBuilder);
107                arguments.add("-p"); // BW project path
108                arguments.add(buildSrcDirectory.getAbsolutePath());
109                arguments.add("-o"); // output file
110                arguments.add(outputFile.getAbsolutePath());
111                arguments.add("-x"); // overwrite the output
112                if (!hideLibraryResources) {
113                        arguments.add("-v"); // validate the project
114                }
115
116                getLog().info(BUILDING_PROJLIB);
117
118                ArrayList<File> tras = new ArrayList<File>();
119                tras.add(tibcoBuildLibraryTRAPath);
120                if (tibcoBuildLibraryUseDesignerTRA) {
121                        tras.add(tibcoDesignerTRAPath);
122                }
123                launchTIBCOBinary(tibcoBuildLibraryPath, tras, arguments, directory, BUILD_PROJLIB_FAILED);
124        }
125
126        public void execute() throws MojoExecutionException {
127        if (skipCompile || skipProjlibCompile) {
128                getLog().info(SKIPPING);
129                
130            File outputFile = getOutputFile();
131                if (outputFile != null && !outputFile.exists() && touchProjlibIfSkipped) {
132                        // Projlib was not created because compilation is skipped
133                        // however we "touch" the Projlib file so there is an empty Projlib file created
134                        try {
135                                outputFile.getParentFile().mkdirs();
136                                        outputFile.createNewFile();
137                                } catch (IOException e) {
138                                        throw new MojoExecutionException(e.getLocalizedMessage(), e);
139                                }
140                }
141            if (outputFile != null && outputFile.exists()) {
142                attachArtifact(outputFile);
143            } else {
144                getLog().warn(WARN_NO_ARTIFACT_ATTACHED);
145            }
146                return;
147        }
148
149                if (isCurrentGoal("bw:launch-designer")) {
150                        return; // ignore
151                }
152
153                super.execute();
154
155                File outputFile = getOutputFile();
156                
157                try {
158                        buildProjlib(outputFile);
159                } catch (IOException e) {
160                        throw new MojoExecutionException(BUILD_PROJLIB_FAILED, e);
161                }
162                
163                attachArtifact(outputFile);
164        }
165
166    private void attachArtifact(File outputFile) {
167        getProject().getArtifact().setFile(outputFile);
168    }
169}
